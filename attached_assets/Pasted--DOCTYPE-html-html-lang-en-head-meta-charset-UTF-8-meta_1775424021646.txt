      const container = document.getElementById('trusted-list');
      container.innerHTML = '';
      landscapers.forEach(l => {
        const isTrusted = l.rating >= 4.7 && l.jobs >= 50;
        const html = `
          <div onclick="startBooking()" class="card rounded-3xl p-5 flex gap-4 cursor-pointer btn-hover micro-tap">
            <div class="text-5xl flex-shrink-0">🌿</div>
            <div class="flex-1">

              </div>
              <p class="text-xs text-white">${l.rating} ★ • ${l.jobs} jobs completed</p>
              <button class="mt-3 bg-[#34C759] text-black px-6 py-2 rounded-3xl text-sm font-medium btn-hover">Book Now</button>
            </div>
          </div>`;
        container.innerHTML += html;
      });
    }

    function switchTab(tab) {
      document.querySelectorAll('#screen > div').forEach(d => d.classList.add('hidden'));
      const screens = ['home-screen','search-screen','appointments-screen','myjobs-screen','profile-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');
      document.querySelectorAll('.nav-item').forEach((item,i) => i === tab ? item.classList.add('active','text-[#34FF7A]') : item.classList.remove('active','text-[#34FF7A]'));
    }
