  <script>
    // Simulated DB (localStorage)
    let db = { users: {}, jobs: [], landscaperPrices: { "John Rivera Landscaping": { "Lawn Mowing": {Small:45,Medium:65,Large:120}, "Hedge Trimming":{Small:55,Medium:75,Large:95}, "Mulching":{Small:110,Medium:140,Large:180}, "Clean Up":{Small:30,Medium:40,Large:55} } } };
    let currentUserType = null;
    let selectedService = '';
    let selectedYardSize = '';
    let selectedLandscaper = "John Rivera Landscaping";
    let currentTip = 0;

    function loadDB() { const saved = localStorage.getItem('thelawn_db'); if(saved) db = JSON.parse(saved); }
    function saveDB() { localStorage.setItem('thelawn_db', JSON.stringify(db)); }

    // REAL-TIME NOTIFICATIONS
    function pushNotification(title, message) {
      const list = document.getElementById('notification-list');
      const html = `<div class="bg-[#222222] rounded-3xl p-4 flex gap-3"><div class="text-3xl">🛎️</div><div><p class="font-medium text-white">${title}</p><p class="text-xs text-white/70">${message}</p></div></div>`;
      list.innerHTML = html + list.innerHTML;
      // Auto-show panel for demo
      document.getElementById('notification-panel').classList.remove('hidden');
      setTimeout(() => { if(Math.random()>0.7) pushNotification("Landscaper Arrived", "John Rivera has arrived at your location"); }, 8000);
    }

    function toggleNotifications() {
      const panel = document.getElementById('notification-panel');
      panel.classList.toggle('hidden');
      if(!panel.classList.contains('hidden')) pushNotification("New Job Update", "Your Lawn Mowing job is now In Progress");
    }
    function hideNotifications() { document.getElementById('notification-panel').classList.add('hidden'); }

    // OPTIMIZED BOOKING FLOW
    function startBooking() {
      selectedService = ''; selectedYardSize = ''; currentTip = 0;
      document.getElementById('home-screen').classList.add('hidden');
      document.getElementById('service-selection-screen').classList.remove('hidden');
      renderServiceCards();
    }

    function selectService(service) {
      selectedService = service;
      document.getElementById('service-selection-screen').classList.add('hidden');
      document.getElementById('job-details-screen').classList.remove('hidden');
      renderYardSizeOptions();
    }

    function renderYardSizeOptions() {
      const container = document.getElementById('yard-size-options');
      container.innerHTML = `
        <button onclick="selectYardSize(this, 'Small')" class="yard-btn bg-[#111111] border border-[#222222] rounded-3xl p-4 text-center text-sm">Small<br><span class="text-xs text-white/70">&lt; 5,000 sq ft</span></button>
        <button onclick="selectYardSize(this, 'Medium')" class="yard-btn bg-[#111111] border border-[#222222] rounded-3xl p-4 text-center text-sm">Medium<br><span class="text-xs text-white/70">5,000–10,000 sq ft</span></button>
        <button onclick="selectYardSize(this, 'Large')" class="yard-btn bg-[#111111] border border-[#222222] rounded-3xl p-4 text-center text-sm">Large<br><span class="text-xs text-white/70">10,000+ sq ft</span></button>`;
    }

    function selectYardSize(btn, size) {
      selectedYardSize = size;
      document.querySelectorAll('.yard-btn').forEach(b => b.classList.remove('border-[#34FF7A]', 'bg-[#34FF7A]', 'text-black'));
      btn.classList.add('border-[#34FF7A]', 'bg-[#34FF7A]', 'text-black');
      document.getElementById('continue-pay-btn').disabled = false;
    }

    function goToPay() {
      const prices = db.landscaperPrices[selectedLandscaper] || {};
      const basePrice = prices[selectedService] ? prices[selectedService][selectedYardSize] || 45 : 45;
      currentTip = 0;

      document.getElementById('service-name-display').textContent = selectedService;
      document.getElementById('subtotal-display').textContent = `$${basePrice}.00`;
      document.getElementById('tip-display').textContent = `$0.00`;
      document.getElementById('total-display').textContent = `$${basePrice}.00`;
      document.getElementById('final-total').textContent = `$${basePrice}.00`;

      // Save job
      db.jobs.push({ id: Date.now(), customer: "Zamire Smith", service: selectedService, yardSize: selectedYardSize, price: basePrice, status: "Scheduled", address: document.getElementById('service-address').value || "8910 45th Ave E, Ellenton, FL" });
      saveDB();

      document.getElementById('job-details-screen').classList.add('hidden');
      document.getElementById('pay-screen').classList.remove('hidden');
    }

    // CUSTOM TIP + NO TIP
    function showCustomTip() {
      document.getElementById('custom-tip-input').classList.remove('hidden');
    }
    function applyCustomTip() {
      const amount = parseFloat(document.getElementById('custom-tip-amount').value) || 0;
      currentTip = amount;
      document.getElementById('tip-display').textContent = `$${amount.toFixed(2)}`;
      const subtotal = parseFloat(document.getElementById('subtotal-display').textContent.replace('$',''));
      const total = subtotal + amount;
      document.getElementById('total-display').textContent = `$${total.toFixed(2)}`;
      document.getElementById('final-total').textContent = `$${total.toFixed(2)}`;
      document.getElementById('custom-tip-input').classList.add('hidden');
    }
    function selectNoTip() {
      currentTip = 0;
      document.getElementById('tip-display').textContent = `$0.00`;
      const subtotal = parseFloat(document.getElementById('subtotal-display').textContent.replace('$',''));
      document.getElementById('total-display').textContent = `$${subtotal.toFixed(2)}`;
      document.getElementById('final-total').textContent = `$${subtotal.toFixed(2)}`;
    }
    function selectTip(el) {
      document.querySelectorAll('#pay-screen button:not([onclick^="selectNoTip"])').forEach(b => b.classList.remove('bg-[#34FF7A]', 'text-black'));
      el.classList.add('bg-[#34FF7A]', 'text-black');
      const percent = parseFloat(el.textContent);
      const subtotal = parseFloat(document.getElementById('subtotal-display').textContent.replace('$',''));
      currentTip = subtotal * (percent / 100);
      document.getElementById('tip-display').textContent = `$${currentTip.toFixed(2)}`;
      document.getElementById('total-display').textContent = `$${(subtotal + currentTip).toFixed(2)}`;
      document.getElementById('final-total').textContent = `$${(subtotal + currentTip).toFixed(2)}`;
      document.getElementById('custom-tip-input').classList.add('hidden');
    }

    function authorizeEscrowHold() {
      alert("✅ Payment held in escrow – both parties must confirm completion.");
      if(db.jobs.length) db.jobs[db.jobs.length-1].status = "Completed";
      saveDB();
      pushNotification("Payment Authorized", "Your escrow payment for " + selectedService + " has been secured.");
      switchTab(3);
    }

    function renderJobStatusTracking() {
      const container = document.getElementById('job-status-list');
      container.innerHTML = db.jobs.map(job => `
        <div class="card rounded-3xl p-6">
          <div class="flex justify-between items-center">
            <p class="font-semibold text-white">${job.service} – ${job.customer}</p>
            <span class="status-pill ${job.status === 'Completed' ? 'bg-[#34FF7A] text-black' : 'bg-[#FF9500] text-white'}">${job.status}</span>
          </div>
          <p class="text-xs text-white/70 mt-1">${job.address} • ${job.yardSize} yard</p>
          <p class="text-xs text-white/70">$${job.price}</p>
        </div>`).join('');
    }

    function switchTab(tab) {
      document.querySelectorAll('#screen > div').forEach(d => d.classList.add('hidden'));
      const screens = ['home-screen','search-screen','appointments-screen','myjobs-screen','profile-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');
      document.querySelectorAll('.nav-item').forEach((item,i) => i === tab ? item.classList.add('active','text-[#34FF7A]') : item.classList.remove('active','text-[#34FF7A]'));
    }

    function goBackToHome() {
      document.getElementById('service-selection-screen').classList.add('hidden');
      document.getElementById('home-screen').classList.remove('hidden');
    }

    function goBack() {
      const current = document.querySelector('#screen > div:not(.hidden)');
      if(current.id === 'pay-screen') { document.getElementById('pay-screen').classList.add('hidden'); document.getElementById('job-details-screen').classList.remove('hidden'); }
      else if(current.id === 'job-details-screen') { document.getElementById('job-details-screen').classList.add('hidden'); document.getElementById('service-selection-screen').classList.remove('hidden'); }
      else switchTab(0);
    }

    window.onload = () => {
      loadDB();
      console.log("%c✅ COMPLETE The Lawn App – Real-Time Notifications + Optimized Booking Flow + Custom/No Tip + Black Review & Pay 🌱", "color:#34FF7A;font-weight:bold");
      // Demo real-time notifications
      setTimeout(() => { if(document.getElementById('app-container').classList.contains('hidden') === false) pushNotification("Landscaper Update", "John Rivera has arrived at your property"); }, 4000);
      renderJobStatusTracking();
    };
  </script>
