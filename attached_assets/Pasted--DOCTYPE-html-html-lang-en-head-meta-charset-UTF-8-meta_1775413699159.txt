  <script>
    function goToBooking() { document.getElementById('home').classList.add('hidden'); document.getElementById('booking').classList.remove('hidden'); }
    function goToPay() { document.getElementById('booking').classList.add('hidden'); document.getElementById('pay').classList.remove('hidden'); }
    function goToTimer() { document.getElementById('pay').classList.add('hidden'); document.getElementById('timer').classList.remove('hidden'); }
    function goToFeedback() { document.getElementById('timer').classList.add('hidden'); document.getElementById('feedback').classList.remove('hidden'); }
    function goBack() {
      const screens = ['home','booking','pay','timer','feedback'];
      for (let s of screens) document.getElementById(s).classList.add('hidden');
      document.getElementById('home').classList.remove('hidden');
    }
    function selectService(el) {
      document.querySelectorAll('#booking > div.grid > div').forEach(d => d.classList.remove('border-[#34C759]'));
      el.classList.add('border-[#34C759]');
    }
