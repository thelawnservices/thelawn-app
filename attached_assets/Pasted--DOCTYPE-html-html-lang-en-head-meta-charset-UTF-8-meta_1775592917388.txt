  <script>
    let db = { 
      users: {}, 
      jobs: [], 
      requests: [], 
      notifications: [], 
      reviews: [],
      servicePhotos: [],
      landscaperPrices: {} 
    };
    let currentUserType = null;
    let currentUser = null;
    let currentRequestFiles = [];
    let currentChatJobId = null;

    function loadDB() { const saved = localStorage.getItem('thelawn_db'); if(saved) db = JSON.parse(saved); }
    function saveDB() { localStorage.setItem('thelawn_db', JSON.stringify(db)); }

    /* ==================== LANDSCAPER REGISTRATION – SERVICE SELECTION & PRICING ==================== */
    function completeLandscaperRegistration() {
      const businessName = document.getElementById('lreg-business-name').value.trim();
      const email = document.getElementById('lreg-email').value.trim();
      const pass = document.getElementById('lreg-password').value.trim();
      const phone = document.getElementById('lreg-phone').value.trim();
      const state = document.getElementById('lreg-state').value.trim();
      const zip = document.getElementById('lreg-zip').value.trim();
      const paymentPref = document.getElementById('lreg-payment-pref').value;

      // NEW: Selected services
      const selectedServices = [];
      const serviceCheckboxes = document.querySelectorAll('.service-checkbox:checked');
      serviceCheckboxes.forEach(cb => selectedServices.push(cb.value));

      if (!businessName || !email || !pass || !phone || !state || !zip || !paymentPref || selectedServices.length === 0) {
        alert("Please fill all fields and select at least one service");
        return;
      }

      // Price inputs per service
      const prices = {};
      selectedServices.forEach(svc => {
        prices[svc] = {
          Small: parseFloat(document.getElementById(`price-${svc}-small`).value) || 0,
          Medium: parseFloat(document.getElementById(`price-${svc}-medium`).value) || 0,
          Large: parseFloat(document.getElementById(`price-${svc}-large`).value) || 0
        };
      });

      currentUser = { 
        businessName, email, password: pass, phone, state, zipcode: zip, 
        paymentPreference: paymentPref, type: 'landscaper', 
        profilePicture: null, heroBackground: null,
        selectedServices, prices 
      };
      currentUserType = 'landscaper';
      document.getElementById('landscaper-register-screen').classList.add('hidden');
      document.getElementById('home-screen').classList.remove('hidden');
      updateHomeScreen();
      updateProfileIcon();
      setTimeout(() => showPasskeyModal(), 600);
      saveDB();
    }

    /* ==================== LANDSCAPER PROFILE – FULLY EDITABLE ==================== */
    function viewProfile() {
      toggleProfileDropdown();
      if (currentUserType === 'landscaper') {
        document.querySelectorAll('.iphone > div[id$="-screen"]').forEach(d => d.classList.add('hidden'));
        document.getElementById('landscaper-profile-screen').classList.remove('hidden');

        // Populate hero
        document.getElementById('landscaper-hero').style.backgroundImage = currentUser.heroBackground ? `url(${currentUser.heroBackground})` : "url('https://picsum.photos/id/1015/800/400')";
        document.getElementById('landscaper-avatar-img').src = currentUser.profilePicture || "https://picsum.photos/id/1015/300/300";
        document.getElementById('landscaper-name').textContent = currentUser.businessName;
        document.getElementById('landscaper-established').textContent = `Est. 2018 • ${currentUser.state || 'FL'}`;
        document.getElementById('edit-background-btn').classList.toggle('hidden', false);

        // Populate services tab (only selected ones)
        const servicesContainer = document.getElementById('services-list');
        servicesContainer.innerHTML = '';
        (currentUser.selectedServices || []).forEach(svc => {
          const prices = currentUser.prices[svc] || {};
          const div = document.createElement('div');
          div.className = "card rounded-3xl p-5";
          div.innerHTML = `
            <p class="font-semibold">${svc}</p>
            <div class="grid grid-cols-3 gap-4 mt-4 text-sm">
              <div>Small: <span class="font-bold">$${prices.Small || 0}</span></div>
              <div>Medium: <span class="font-bold">$${prices.Medium || 0}</span></div>
              <div>Large: <span class="font-bold">$${prices.Large || 0}</span></div>
            </div>
          `;
          servicesContainer.appendChild(div);
        });

        // Photos of work
        renderServicePhotos();
      } else {
        switchTab(4);
      }
    }

    function changeProfilePicture() {
      if (currentUserType !== 'landscaper') return;
      const input = document.createElement('input');
      input.type = 'file';
      input.accept = 'image/*';
      input.onchange = e => {
        const file = e.target.files[0];
        const reader = new FileReader();
        reader.onload = ev => {
          currentUser.profilePicture = ev.target.result;
          document.getElementById('landscaper-avatar-img').src = ev.target.result;
          saveDB();
        };
        reader.readAsDataURL(file);
      };
      input.click();
    }

    function changeHeroBackground() {
      if (currentUserType !== 'landscaper') return;
      const input = document.createElement('input');
      input.type = 'file';
      input.accept = 'image/*';
      input.onchange = e => {
        const file = e.target.files[0];
        const reader = new FileReader();
        reader.onload = ev => {
          currentUser.heroBackground = ev.target.result;
          document.getElementById('landscaper-hero').style.backgroundImage = `url(${ev.target.result})`;
          saveDB();
        };
        reader.readAsDataURL(file);
      };
      input.click();
    }

    /* ==================== CUSTOMER REVIEW ON LANDSCAPER PROFILE ==================== */
    function leaveReviewFromProfile() {
      if (currentUserType !== 'customer') return;
      document.getElementById('landscaper-reviews-tab').classList.remove('hidden');
      document.getElementById('landscaper-info-tab').classList.add('hidden');
      document.getElementById('landscaper-services-tab').classList.add('hidden');
    }

    function submitReview() {
      const text = document.getElementById('new-review-text').value.trim();
      if (!text) return alert("Please write a review");
      
      if (!db.reviews) db.reviews = [];
      db.reviews.push({
        landscaper: currentUser.businessName,
        reviewer: currentUser.first + " " + currentUser.last,
        text,
        date: new Date().toLocaleDateString()
      });
      saveDB();
      alert("✅ Review posted!");
      document.getElementById('new-review-text').value = '';
      renderReviews();
    }

    function renderReviews() {
      const container = document.getElementById('reviews-list');
      container.innerHTML = '';
      (db.reviews || []).filter(r => r.landscaper === currentUser.businessName).forEach(review => {
        const div = document.createElement('div');
        div.className = "card rounded-3xl p-5";
        div.innerHTML = `<p class="font-medium">${review.reviewer}</p><p class="text-white/80">${review.text}</p><p class="text-xs text-white/50">${review.date}</p>`;
        container.appendChild(div);
      });
    }

    /* ==================== BOOK FROM PROFILE ==================== */
    function startBookingFromProfile() {
      alert("📅 Starting booking process for " + currentUser.businessName);
      document.getElementById('landscaper-profile-screen').classList.add('hidden');
      document.getElementById('availability-screen').classList.remove('hidden');
      renderServiceTypeButtons();
    }

    /* ==================== SERVICE PHOTOS UPLOAD (Landscaper only) ==================== */
    function uploadServicePhotos() {
      const input = document.createElement('input');
      input.type = 'file';
      input.accept = 'image/*';
      input.multiple = true;
      input.onchange = e => {
        Array.from(e.target.files).forEach(file => {
          const reader = new FileReader();
          reader.onload = ev => {
            if (!db.servicePhotos) db.servicePhotos = [];
            db.servicePhotos.push({ landscaper: currentUser.businessName, url: ev.target.result });
            saveDB();
            renderServicePhotos();
          };
          reader.readAsDataURL(file);
        });
      };
      input.click();
    }

    function renderServicePhotos() {
      const grid = document.getElementById('service-photos-grid');
      grid.innerHTML = '';
      (db.servicePhotos || []).filter(p => p.landscaper === currentUser.businessName).forEach(photo => {
        const div = document.createElement('div');
        div.className = "aspect-video bg-[#222222] rounded-3xl overflow-hidden";
        div.innerHTML = `<img src="${photo.url}" class="w-full h-full object-cover">`;
        grid.appendChild(div);
      });
    }
