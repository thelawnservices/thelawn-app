    function switchTab(tab) {
      const currentScreen = document.querySelector('#screen > div:not(.hidden)');
      if (currentScreen) {
        currentScreen.classList.add(tab > parseInt(currentScreen.id.split('-')[0].slice(-1)) ? 'screen-exit-right' : 'screen-exit-left');
        setTimeout(() => {
          document.querySelectorAll('#screen > div').forEach(d => d.classList.add('hidden'));
          const target = document.getElementById(['home-screen','search-screen','appointments-screen','myjobs-screen','profile-screen'][tab]);
          target.classList.remove('hidden');
          target.classList.add('screen-enter');
          document.querySelectorAll('.nav-item').forEach((item,i) => i === tab ? item.classList.add('active','text-[#34FF7A]') : item.classList.remove('active','text-[#34FF7A]'));
