  <script>
    let selectedTime = null;

    function startBooking() {
      hideAll();
      document.getElementById('availability-screen').classList.remove('hidden');
      renderTimeSlots();
    }

    function renderTimeSlots() {
      const slotsHTML = `
        <div onclick="selectTimeSlot(this)" class="time-slot bg-white border border-gray-200 rounded-2xl py-4 text-center text-sm font-medium cursor-pointer">9:00 AM</div>
        <div onclick="selectTimeSlot(this)" class="time-slot bg-white border border-gray-200 rounded-2xl py-4 text-center text-sm font-medium cursor-pointer">10:30 AM</div>
        <div onclick="selectTimeSlot(this)" class="time-slot bg-white border border-gray-200 rounded-2xl py-4 text-center text-sm font-medium cursor-pointer">1:00 PM</div>
        <div onclick="selectTimeSlot(this)" class="time-slot bg-white border border-gray-200 rounded-2xl py-4 text-center text-sm font-medium cursor-pointer">2:30 PM</div>
      `;
      document.getElementById('time-slots').innerHTML = slotsHTML;
    }

    function selectTimeSlot(el) {
      document.querySelectorAll('.time-slot').forEach(s => s.classList.remove('bg-[#34C759]', 'text-white'));
      el.classList.add('bg-[#34C759]', 'text-white');
      selectedTime = el.textContent;
      document.getElementById('continue-btn').disabled = false;
      document.getElementById('continue-btn').classList.remove('opacity-50');
    }

    function goToJobDetails() {
      hideAll();
      document.getElementById('job-details-screen').classList.remove('hidden');
    }

    function goToPay() {
      hideAll();
      document.getElementById('pay-screen').classList.remove('hidden');
    }

    function authorizeEscrowHold() {
      hideAll();
      document.getElementById('success-screen').classList.remove('hidden');
    }

    function goToMyJobs() {
      hideAll();
      document.getElementById('myjobs-screen').classList.remove('hidden');
      switchTab(3);
    }

    function hideAll() {
      document.querySelectorAll('#screen > div').forEach(d => d.classList.add('hidden'));
    }

    function switchTab(tab) {
      hideAll();
      const screens = ['home-screen','search-screen','appointments-screen','myjobs-screen','profile-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');
      document.querySelectorAll('.nav-item').forEach((item,i) => i === tab ? item.classList.add('active','text-[#34C759]') : item.classList.remove('active','text-[#34C759]'));
    }

    function toggleDeviceMode() {
      const c = document.getElementById('preview-container');
      c.classList.toggle('tablet');
      document.getElementById('mode-badge').textContent = c.classList.contains('tablet') ? 'TABLET' : 'PHONE';
    }

    function toggleNotifications() {
      alert("🛎️ Notifications panel would open here.");
    }

    window.onload = () => {
      console.log("%c✅ COMPLETE The Lawn App – Book Now → Availability Selection (based on landscaper availability) 🌱", "color:#34C759;font-weight:bold");
      switchTab(0);
    };
  </script>
