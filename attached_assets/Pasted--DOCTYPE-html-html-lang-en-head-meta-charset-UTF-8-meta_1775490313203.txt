  <script>
    let db = { users: {}, jobs: [], requests: [], landscaperPrices: { "John Rivera Landscaping": { "Lawn Mowing": {Small:45,Medium:65,Large:120}, "Hedge Trimming":{Small:55,Medium:75,Large:95}, "Mulching":{Small:110,Medium:140,Large:180}, "Clean Up":{Small:30,Medium:40,Large:55} } } };
    let currentUserType = null;
    let currentUser = null;

    function loadDB() { const saved = localStorage.getItem('thelawn_db'); if(saved) db = JSON.parse(saved); }
    function saveDB() { localStorage.setItem('thelawn_db', JSON.stringify(db)); }

    function updateProfileIcon() {
      const icon = document.getElementById('profile-icon');
      if (!currentUser) return;
      if (currentUser.profilePicture) {
        icon.innerHTML = `<img src="${currentUser.profilePicture}" class="w-full h-full object-cover">`;
      } else {
        let letter = currentUserType === 'customer' && currentUser.first ? currentUser.first[0].toUpperCase() : currentUser.businessName ? currentUser.businessName[0].toUpperCase() : '';
        icon.innerHTML = letter ? `<span class="text-xl font-bold">${letter}</span>` : '👤';
      }
    }

    function changeProfilePicture() {
      if (confirm("iOS would now ask for permission to access your Photos.\n\nAllow access to continue?")) {
        const input = document.createElement('input');
        input.type = 'file';
        input.accept = 'image/*';
        input.onchange = e => {
          const file = e.target.files[0];
          if (file) {
            const reader = new FileReader();
            reader.onload = ev => {
              currentUser.profilePicture = ev.target.result;
              updateProfileIcon();
              if (currentUserType === 'landscaper') {
                document.getElementById('large-profile-pic').innerHTML = `<img src="${ev.target.result}" class="w-full h-full object-cover">`;
              } else {
                document.getElementById('customer-large-pic').innerHTML = `<img src="${ev.target.result}" class="w-full h-full object-cover">`;
              }
              saveDB();
              alert("✅ Profile picture updated!");
            };
            reader.readAsDataURL(file);
          }
        };
        input.click();
      }
    }

    function toggleProfileDropdown() {
      const dropdown = document.getElementById('profile-dropdown');
      dropdown.classList.toggle('hidden');
      document.getElementById('availability-dropdown-item').classList.toggle('hidden', currentUserType !== 'landscaper');
    }

    function openSettings() { toggleProfileDropdown(); document.getElementById('settings-modal').classList.remove('hidden'); }
    function closeSettings() { document.getElementById('settings-modal').classList.add('hidden'); }

    function grantPushPermission() {
      closePushModal();
      alert("✅ Push notifications enabled! You will now receive real-time alerts for requests, appointments, and updates.");
      // In a real app this would call Notification.requestPermission()
    }
    function closePushModal() { document.getElementById('push-permission-modal').classList.add('hidden'); }

    function askForPushPermission() {
      if (!localStorage.getItem('pushAsked')) {
        document.getElementById('push-permission-modal').classList.remove('hidden');
        localStorage.setItem('pushAsked', 'true');
      }
    }

    function switchTab(tab) {
      document.querySelectorAll('.iphone > div[id$="-screen"]').forEach(d => d.classList.add('hidden'));
      const screens = ['home-screen','search-screen','appointments-screen','myjobs-screen','profile-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');
      if (tab === 4) updateProfileScreen();
    }

    function updateProfileScreen() {
      if (currentUserType === 'landscaper') {
        document.getElementById('landscaper-profile').classList.remove('hidden');
        document.getElementById('customer-profile').classList.add('hidden');
        document.getElementById('business-name-display').textContent = currentUser.businessName || "Your Business";
      } else {
        document.getElementById('landscaper-profile').classList.add('hidden');
        document.getElementById('customer-profile').classList.remove('hidden');
        document.getElementById('customer-location-display').textContent = currentUser.address || "No address set";
      }
    }
