  <script>
    let isOffline = false;

    function toggleOffline() {
      isOffline = !isOffline;
      const banner = document.getElementById('offline-banner');
      const btn = document.getElementById('offline-btn');
      if (isOffline) {
        banner.classList.remove('hidden');
        btn.innerHTML = 'Offline';
        btn.classList.add('bg-[#FF3B30]');
      } else {
        banner.classList.add('hidden');
        btn.innerHTML = 'Online';
        btn.classList.remove('bg-[#FF3B30]');
      }
    }

    function loginAs(type) {
      document.getElementById('login-screen').classList.add('hidden');
      document.getElementById('app-container').classList.remove('hidden');
      if (type === 'landscaper') alert("Logged in as Landscaper");
    }

    function showNotifications() {
      alert("🛎️ Push Notifications\n\n• John Rivera is on the way!\n• New trusted landscaper nearby");
    }

    function renderHorizontalTrusted() {
      const container = document.getElementById('trusted-horizontal');
      container.innerHTML = '';
      const topLandscapers = [
        { name: "John Rivera Landscaping", rating: 4.9, jobs: 142 },
        { name: "Sarah’s Lawn Care", rating: 5.0, jobs: 98 },
        { name: "GreenScape Pros", rating: 4.8, jobs: 87 },
        { name: "Elite Lawn Services", rating: 4.9, jobs: 65 },
        { name: "FreshCut Landscaping", rating: 5.0, jobs: 112 },
        { name: "Premier Turf Care", rating: 4.7, jobs: 79 },
        { name: "Nature’s Edge Lawn", rating: 4.9, jobs: 53 }
      ];
      topLandscapers.forEach(l => {
        const html = `<div onclick="startBooking()" class="card flex-shrink-0 w-52 rounded-3xl p-5 cursor-pointer btn-hover micro-tap">…</div>`;
        container.innerHTML += html;
      });
    }

    function switchTab(tab) {
      document.querySelectorAll('#screen > div').forEach(d => d.classList.add('hidden'));
      const screens = ['home-screen','search-screen','appointments-screen','myjobs-screen','profile-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');
      document.querySelectorAll('.nav-item').forEach((item,i) => i === tab ? item.classList.add('active','text-[#34FF7A]') : item.classList.remove('active','text-[#34FF7A]'));
    }

    function startBooking() {
      if (isOffline) {
        alert("❌ You are offline. Booking is disabled.");
        return;
      }
      switchTab(2);
    }

    window.onload = () => {
      console.log("%c✅ COMPLETE TheLawn App – Offline/Online + Refined Login Animations + Push Notifications + Payment Animations + Mobile CSS Optimization 🌱", "color:#34FF7A;font-weight:bold");
      renderHorizontalTrusted();
      switchTab(0);
    };
