    // NOTIFICATIONS – disable feature + badge
    function toggleNotificationSetting() {
      notificationsEnabled = !notificationsEnabled;
      const statusEl = document.getElementById('notif-status');
      statusEl.textContent = notificationsEnabled ? "ON" : "OFF";
      updateNotificationBadge();
    }

    function updateNotificationBadge() {
      const bell = document.querySelector('.header span[onclick="toggleNotifications()"]');
      if (!bell) return;
      if (notificationsEnabled) {
        bell.innerHTML = `🛎️`;
        bell.onclick = toggleNotifications;
      } else {
        bell.innerHTML = `🛎️<span class="absolute -top-1 -right-1 bg-[#FF3B30] text-white text-[10px] font-bold rounded-full h-4 w-4 flex items-center justify-center">${notificationCount}</span>`;
        bell.onclick = () => alert("Notifications are turned OFF");
      }
    }

    function pushNotification(title, message) {
      if (!notificationsEnabled) {
        notificationCount++;
        updateNotificationBadge();
        return;
      }
      const list = document.getElementById('notification-list');
      const html = `<div class="refined-notif rounded-3xl p-4 flex gap-3"><div class="text-3xl">🛎️</div><div><p class="font-medium text-white">${title}</p><p class="text-xs text-white/70">${message}</p></div></div>`;
      list.innerHTML = html + list.innerHTML;
      document.getElementById('notification-panel').classList.remove('hidden');
    }

    function toggleNotifications() {
      if (!notificationsEnabled) return;
      const panel = document.getElementById('notification-panel');
      panel.classList.toggle('hidden');
    }
    function hideNotifications() { document.getElementById('notification-panel').classList.add('hidden'); }

    // PAYMENT DROPDOWN IN REVIEW & PAY
    function handlePaymentMethodChange() {
      const method = document.getElementById('payment-method').value;
      const container = document.getElementById('payment-fields');
      container.innerHTML = '';

      if (method === 'applepay') {
        container.innerHTML = `<div class="bg-[#111111] rounded-3xl p-4 text-center text-[#34FF7A]">✅ Apple Pay ready – tap to confirm</div>`;
      } else if (method === 'debit') {
        container.innerHTML = `
          <input type="text" placeholder="Card Number" class="w-full bg-[#111111] border border-[#222222] rounded-3xl p-4 text-white mb-3">
          <div class="grid grid-cols-2 gap-3">
            <input type="text" placeholder="MM/YY" class="bg-[#111111] border border-[#222222] rounded-3xl p-4 text-white">
            <input type="text" placeholder="CVV" class="bg-[#111111] border border-[#222222] rounded-3xl p-4 text-white">
          </div>`;
      } else if (method === 'venmo') {
        container.innerHTML = `<input type="text" placeholder="Venmo Username" class="w-full bg-[#111111] border border-[#222222] rounded-3xl p-4 text-white">`;
      } else if (method === 'paypal') {
        container.innerHTML = `<input type="email" placeholder="PayPal Email" class="w-full bg-[#111111] border border-[#222222] rounded-3xl p-4 text-white">`;
      } else if (method === 'cashapp') {
        container.innerHTML = `<input type="text" placeholder="$Cashtag" class="w-full bg-[#111111] border border-[#222222] rounded-3xl p-4 text-white">`;
      }
    }

    function goToPay() {
      const prices = db.landscaperPrices[selectedLandscaper] || {};
      const basePrice = prices[selectedService] ? prices[selectedService][selectedYardSize] || 45 : 45;
      currentTip = 0;

      document.getElementById('service-name-display').textContent = selectedService;
      if (isRecurring) {
        document.getElementById('recurring-badge').classList.remove('hidden');
        document.getElementById('recurring-note').classList.remove('hidden');
      }

      document.getElementById('subtotal-display').textContent = `$${basePrice}.00`;
      document.getElementById('tip-display').textContent = `$0.00`;
      document.getElementById('total-display').textContent = `$${basePrice}.00`;
      document.getElementById('final-total').textContent = `$${basePrice}.00`;

        id: Date.now(),
        customer: "Zamire Smith",
        service: selectedService,
        yardSize: selectedYardSize,
        price: basePrice,
        status: "Scheduled",
        address: document.getElementById('service-address').value || "8910 45th Ave E, Ellenton, FL",

      });

      document.getElementById('job-details-screen').classList.add('hidden');
      document.getElementById('pay-screen').classList.remove('hidden');
    }
