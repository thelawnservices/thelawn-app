  <script>
    let db = { 
      users: {}, 
      jobs: [], 
      requests: [], 
      notifications: [], 
      landscaperPrices: { "John Rivera Landscaping": { "Lawn Mowing": {Small:45,Medium:65,Large:120}, "Hedge Trimming":{Small:55,Medium:75,Large:95}, "Mulching":{Small:110,Medium:140,Large:180}, "Clean Up":{Small:30,Medium:40,Large:55} } } 
    };
    let currentUserType = null;
    let currentUser = null;
    let currentRequestFiles = [];
    let currentChatJobId = null;

    function loadDB() { const saved = localStorage.getItem('thelawn_db'); if(saved) db = JSON.parse(saved); }
    function saveDB() { localStorage.setItem('thelawn_db', JSON.stringify(db)); }

    /* RESTRICTED NOTIFICATIONS – ONLY active services, arrived, request acceptance, completion */
    function sendRestrictedNotification(type, message, jobId) {
      const allowedTypes = ['active_service', 'arrived', 'request_accepted', 'completion'];
      if (!allowedTypes.includes(type)) return;

      const notif = {
        id: Date.now(),
        type,
        message,
        jobId,
        timestamp: new Date().toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'}),
        read: false
      };
      db.notifications.unshift(notif);
      saveDB();
      // Visual badge on bell (simple count)
      console.log(`🛎️ Restricted Notification: ${message}`);
    }

    function toggleNotifications() {
      const panel = document.getElementById('notifications-panel');
      const list = document.getElementById('notifications-list');
      list.innerHTML = '';

      const filtered = db.notifications.filter(n => 
        ['active_service', 'arrived', 'request_accepted', 'completion'].includes(n.type)
      );

      if (filtered.length === 0) {
        list.innerHTML = `<p class="text-white/50 text-center py-12">No active service notifications yet</p>`;
      } else {
        filtered.forEach(n => {
          const div = document.createElement('div');
          div.className = `card rounded-3xl p-5 flex items-start gap-4 ${n.read ? 'opacity-60' : ''}`;
          div.innerHTML = `
            <div class="text-3xl">${n.type === 'request_accepted' ? '✅' : n.type === 'arrived' ? '🚪' : n.type === 'active_service' ? '🛠️' : '🎉'}</div>
            <div class="flex-1">
              <p class="font-medium">${n.message}</p>
              <p class="text-xs text-white/60">${n.timestamp}</p>
            </div>
          `;
          list.appendChild(div);
        });
      }
      panel.classList.toggle('hidden');
    }

    /* CHAT & CALL FOR ACTIVE JOBS (both accounts) */
    function openChat(jobId) {
      currentChatJobId = jobId;
      const job = db.jobs.find(j => j.id === jobId);
      if (!job) return;

      document.getElementById('chat-header').innerHTML = `
        Chat with ${currentUserType === 'customer' ? job.landscaper || 'Landscaper' : job.customer || 'Customer'}
      `;
      document.getElementById('chat-messages').innerHTML = `
        <div class="chat-bubble bg-[#222222] ml-auto text-right">Hi, I’m on my way!</div>
        <div class="chat-bubble bg-[#34FF7A] text-black">Great! See you soon.</div>
      `;
      document.getElementById('chat-modal').classList.remove('hidden');
    }

    function sendChatMessage() {
      const input = document.getElementById('chat-input');
      const msg = input.value.trim();
      if (!msg || !currentChatJobId) return;

      const bubble = document.createElement('div');
      bubble.className = `chat-bubble ${currentUserType === 'customer' ? 'bg-[#222222] ml-auto text-right' : 'bg-[#34FF7A] text-black'}`;
      bubble.textContent = msg;
      document.getElementById('chat-messages').appendChild(bubble);
      document.getElementById('chat-messages').scrollTop = 999999;
      input.value = '';

      // Mock reply after 1s
      setTimeout(() => {
        const reply = document.createElement('div');
        reply.className = `chat-bubble ${currentUserType === 'customer' ? 'bg-[#34FF7A] text-black' : 'bg-[#222222]'}`;
        reply.textContent = currentUserType === 'customer' ? "On my way!" : "Thanks!";
        document.getElementById('chat-messages').appendChild(reply);
        document.getElementById('chat-messages').scrollTop = 999999;
      }, 900);
    }

    function closeChat() {
      document.getElementById('chat-modal').classList.add('hidden');
      currentChatJobId = null;
    }

    function simulateCall(jobId) {
      const job = db.jobs.find(j => j.id === jobId);
      if (!job) return;
      const otherParty = currentUserType === 'customer' ? (job.landscaper || 'Landscaper') : (job.customer || 'Customer');
      alert(`📞 Calling ${otherParty}...\n\n(Connected – this is a demo call simulation)`);
    }

    /* UPDATE MY JOBS – ADD CHAT & CALL ONLY FOR ACTIVE SERVICES */
    function updateMyJobsScreen() {
      const container = document.getElementById('job-status-list');
      container.innerHTML = '';

      const myJobs = db.jobs.filter(j => 
        (currentUserType === 'customer' && j.customer === (currentUser.first + " " + currentUser.last)) ||
        (currentUserType === 'landscaper' && j.landscaper === currentUser.businessName)
      );

      if (myJobs.length === 0) {
        container.innerHTML = `<p class="text-white/50 text-center py-12">No active jobs yet</p>`;
        return;
      }

      myJobs.forEach(job => {
        const isActive = ['Future', 'In Progress', 'Active'].includes(job.status);
        const div = document.createElement('div');
        div.className = `card rounded-3xl p-5`;
        div.innerHTML = `
          <div class="flex justify-between items-start">
            <div>
              <p class="font-semibold">${job.service}</p>
              <p class="text-xs text-white/70">${job.date} • ${job.time}</p>
              <p class="text-sm mt-2">${job.status}</p>
            </div>
            <div class="flex gap-3">
              ${isActive ? `
                <button onclick="openChat(${job.id});event.stopImmediatePropagation()" class="w-9 h-9 bg-[#222222] rounded-2xl flex items-center justify-center text-xl">💬</button>
                <button onclick="simulateCall(${job.id});event.stopImmediatePropagation()" class="w-9 h-9 bg-[#34FF7A] text-black rounded-2xl flex items-center justify-center text-xl">📞</button>
              ` : ''}
            </div>
          </div>
        `;
        container.appendChild(div);
      });
    }

    /* REGISTRATION + LOGIN (kept from previous) */

      updateHomeScreen();

      document.getElementById('home-screen').classList.remove('hidden');
      updateHomeScreen();
      updateProfileIcon();
      setTimeout(() => showPasskeyModal(), 600);
    }

    window.onload = () => {
      loadDB();
      console.log("%c✅ The Lawn – Restricted notifications + Chat & Call icons in active jobs (both accounts) complete 🌱", "color:#34FF7A;font-weight:bold");
    };
