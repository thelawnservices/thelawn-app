  <script>
    let db = { 
      users: {}, 
      jobs: [], 
      requests: [], 
      landscaperPrices: { "John Rivera Landscaping": { "Lawn Mowing": {Small:45,Medium:65,Large:120}, "Hedge Trimming":{Small:55,Medium:75,Large:95}, "Mulching":{Small:110,Medium:140,Large:180}, "Clean Up":{Small:30,Medium:40,Large:55} } },
      landscaperAvailability: { "John Rivera Landscaping": { "Lawn Mowing": {days:["Mon","Tue","Wed","Thu","Fri"], start:"08:00", end:"18:00"} } }
    };
    let currentUserType = null;
    let currentUser = null;
    let selectedService = '';
    let selectedYardSize = '';
    let selectedLandscaper = "John Rivera Landscaping";
    let isRequestMode = false;

    function loadDB() { const saved = localStorage.getItem('thelawn_db'); if(saved) db = JSON.parse(saved); }
    function saveDB() { localStorage.setItem('thelawn_db', JSON.stringify(db)); }

    function updateProfileIcon() {
      const icon = document.getElementById('profile-icon');
      if (!currentUser) { icon.innerHTML = '👤'; return; }
      if (currentUser.profilePicture) {
        icon.innerHTML = `<img src="${currentUser.profilePicture}" class="w-full h-full object-cover">`;
      } else {
        let letter = currentUserType === 'customer' && currentUser.first ? currentUser.first[0].toUpperCase() : currentUser.businessName ? currentUser.businessName[0].toUpperCase() : '';
        icon.innerHTML = letter ? `<span class="text-xl font-bold">${letter}</span>` : '👤';
      }
    }

    function toggleProfileDropdown() {
      const dropdown = document.getElementById('profile-dropdown');
      dropdown.classList.toggle('hidden');
      // Service Availability at the TOP for landscapers
      document.getElementById('availability-dropdown-item').classList.toggle('hidden', currentUserType !== 'landscaper');
    }

    function changeProfilePicture() {
      if (confirm("iOS would now ask for permission to access your Photos.\n\nAllow access to continue?")) {
        const input = document.createElement('input');
        input.type = 'file';
        input.accept = 'image/*';
        input.onchange = e => {
          const file = e.target.files[0];
          if (file) {
            const reader = new FileReader();
            reader.onload = ev => {
              currentUser.profilePicture = ev.target.result;
              updateProfileIcon();
              document.getElementById('large-profile-pic').innerHTML = `<img src="${ev.target.result}" class="w-full h-full object-cover">`;
              saveDB();
              alert("✅ Profile picture updated!");
            };
            reader.readAsDataURL(file);
          }
        };
        input.click();
      }
    }

    function usePasskeyLogin() {
      if (confirm("iOS Passkey / Saved PIN detected.\n\nUse saved biometric login?")) {
        alert("✅ Logged in with Passkey / PIN!");
        document.getElementById('landscaper-login-screen').classList.add('hidden');
        document.getElementById('app-container').classList.remove('hidden'); // assuming you have the main app container
        updateProfileIcon();
        switchTab(0);
      }
    }

    function renderNearbyRequests() {
      const container = document.getElementById('nearby-requests-list');
      const filtered = db.requests.filter(r => r.withinRange === true);
      container.innerHTML = filtered.length ? filtered.map(r => `
        <div class="card rounded-3xl p-5">
          <p class="font-medium">${r.service} • ${r.yardSize}</p>
          <p class="text-xs text-white/70">${r.address}</p>
          <div class="flex gap-3 mt-4">
            <button onclick="acceptRequest(${r.id})" class="flex-1 bg-[#22C55E] text-black py-3 rounded-3xl text-sm font-semibold">Accept</button>
            <button onclick="declineRequest(${r.id})" class="flex-1 bg-[#222222] py-3 rounded-3xl text-sm font-semibold">Decline</button>
          </div>
        </div>`).join('') : `<p class="text-white/50 text-center py-8">No requests in your service range yet.</p>`;
    }

    // Registration now captures location and shows nearby recommendations
    function completeCustomerRegistration() {
      // ... previous fields ...
      currentUser.location = { lat: 28.55, lng: -81.35 }; // demo Windermere, FL
      currentUserType = 'customer';
      // Auto-recommend nearby
      document.getElementById('nearby-pros-list').innerHTML = `<div class="text-sm text-white/70">John Rivera Landscaping • 2.4 mi away</div>`;
      updateProfileIcon();
      switchTab(0);
    }

    function completeLandscaperRegistration() {
      // ... previous fields ...
      currentUser.location = { lat: 28.55, lng: -81.35 };
      currentUserType = 'landscaper';
      updateProfileIcon();
      switchTab(0);
    }

    // SwitchTab now hides the "Find a landscaper near you" header for landscapers
    function switchTab(tab) {
      document.querySelectorAll('.iphone > div[id$="-screen"]').forEach(d => d.classList.add('hidden'));
      const screens = ['home-screen','search-screen','appointments-screen','myjobs-screen','profile-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');

      // Remove header for landscapers only
      if (tab === 0 && currentUserType === 'landscaper') {
        document.getElementById('home-header-customer').classList.add('hidden');
      } else if (tab === 0) {
        document.getElementById('home-header-customer').classList.remove('hidden');
      }

      if (tab === 2 && currentUserType === 'landscaper') {
        renderNearbyRequests();
      }
      updateProfileIcon();
    }

    window.onload = () => {
      loadDB();
      console.log("%c✅ The Lawn – FULLY REFINED PROFESSIONAL UI + All new features implemented 🌱", "color:#22C55E;font-weight:bold;font-size:14px");
      updateProfileIcon();
      renderNearbyRequests();
      renderServiceTypeButtons();
    };
  </script>
