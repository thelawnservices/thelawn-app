  <script>
    let currentUserType = null;
    let registeredUsers = {
      customer: null,
      landscaper: null
    };

    function showLogin(type) {
      document.getElementById('auth-screen').classList.add('hidden');
      if (type === 'customer') document.getElementById('customer-login-screen').classList.remove('hidden');
      else document.getElementById('landscaper-login-screen').classList.remove('hidden');
    }

    function showRegister(type) {
      document.getElementById('auth-screen').classList.add('hidden');
      if (type === 'customer') document.getElementById('customer-register-screen').classList.remove('hidden');
      else document.getElementById('landscaper-register-screen').classList.remove('hidden');
    }

    function completeCustomerRegistration() {
      const first = document.getElementById('reg-firstname').value;
      const last = document.getElementById('reg-lastname').value;
      const email = document.getElementById('reg-email').value;
      const pass = document.getElementById('reg-password').value;
      const phone = document.getElementById('reg-phone').value;
      const address = document.getElementById('reg-address').value;

      if (!first || !last || !email || !pass || !phone || !address) {
        alert("Please fill all fields");
        return;
      }

      registeredUsers.customer = { first, last, email, password: pass, phone, address };
      currentUserType = 'customer';
      document.getElementById('customer-register-screen').classList.add('hidden');
      document.getElementById('app-container').classList.remove('hidden');
      switchTab(0);
      alert("✅ Account created! You can now sign in with your email and password.");
    }

    function completeLandscaperRegistration() {
      const email = document.getElementById('lreg-email').value;
      const pass = document.getElementById('lreg-password').value;
      const phone = document.getElementById('lreg-phone').value;
      const specialty = document.getElementById('lreg-specialty').value;
      const city = document.getElementById('lreg-city').value;
      const state = document.getElementById('lreg-state').value;
      const years = document.getElementById('lreg-years').value;
      const routing = document.getElementById('lreg-routing').value;
      const account = document.getElementById('lreg-account').value;

      if (!email || !pass || !phone || !specialty || !city || !state || !years || !routing || !account) {
        alert("Please fill all fields including banking information");
        return;
      }

      registeredUsers.landscaper = { email, password: pass, phone, specialty, city, state, years, routing, account };
      currentUserType = 'landscaper';
      document.getElementById('landscaper-register-screen').classList.add('hidden');
      document.getElementById('app-container').classList.remove('hidden');
      switchTab(0);
      alert("✅ Landscaper account created with banking details!");
    }

    function attemptLogin(type) {
      const emailInput = type === 'customer' ? document.getElementById('login-email-customer').value : document.getElementById('login-email-landscaper').value;
      const passInput = type === 'customer' ? document.getElementById('login-password-customer').value : document.getElementById('login-password-landscaper').value;

      const user = registeredUsers[type];
      if (user && user.email === emailInput && user.password === passInput) {
        currentUserType = type;
        document.getElementById(type + '-login-screen').classList.add('hidden');
        document.getElementById('app-container').classList.remove('hidden');
        switchTab(0);
      } else {
        alert("Invalid email or password");
      }
    }

    function showPasskeyOption() {
      alert("✅ Passkey successfully set up for iPhone Face ID / Touch ID login");
    }

    function toggleProfileMode() {
      const isLandscaper = document.getElementById('profile-mode-label').textContent.includes('Landscaper');
      document.getElementById('landscaper-profile').classList.toggle('hidden', !isLandscaper);
      document.getElementById('customer-profile').classList.toggle('hidden', isLandscaper);
      document.getElementById('profile-mode-label').textContent = isLandscaper ? "Customer View" : "Landscaper View";
    }

    function renderJobStatusTracking() {
      const container = document.getElementById('job-status-list');
      container.innerHTML = `

    }
