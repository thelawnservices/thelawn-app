    function loadDB() {
      const saved = localStorage.getItem('thelawn_db');
      if (saved) db = JSON.parse(saved);
    }

    function saveDB() {
      localStorage.setItem('thelawn_db', JSON.stringify(db));
    }

    // Service selection cards

      selectedService = service;
      document.getElementById('service-selection-screen').classList.add('hidden');
      document.getElementById('job-details-screen').classList.remove('hidden');
      // Reset yard size buttons
      document.querySelectorAll('.yard-btn').forEach(b => b.classList.remove('border-[#34FF7A]', 'bg-[#34FF7A]', 'text-black'));
    }

    function selectYardSize(btn, size) {
      selectedYardSize = size;
      document.querySelectorAll('.yard-btn').forEach(b => b.classList.remove('border-[#34FF7A]', 'bg-[#34FF7A]', 'text-black'));
      btn.classList.add('border-[#34FF7A]', 'bg-[#34FF7A]', 'text-black');
      document.getElementById('continue-pay-btn').disabled = false;
    }

    function goToPay() {
      // Calculate dynamic price from simulated DB
      const prices = db.landscaperPrices[selectedLandscaper] || {};
      const basePrice = prices[selectedService] ? prices[selectedService][selectedYardSize] || 45 : 45;

      // Update Review & Pay screen
      document.getElementById('service-name-display').textContent = selectedService;
      document.getElementById('subtotal-display').textContent = `$${basePrice}.00`;
      document.getElementById('total-display').textContent = `$${(basePrice * 1.15).toFixed(2)}`;
      document.getElementById('final-total').textContent = `$${(basePrice * 1.15).toFixed(2)}`;

      // Save job to simulated DB
      db.jobs.push({
        id: Date.now(),
        customer: "Zamire Smith",
        service: selectedService,
        yardSize: selectedYardSize,
        price: basePrice,
        status: "Scheduled",
        address: document.getElementById('service-address').value || "8910 45th Ave E, Ellenton, FL"
      });
      saveDB();

      document.getElementById('job-details-screen').classList.add('hidden');
      document.getElementById('pay-screen').classList.remove('hidden');
    }

    function authorizeEscrowHold() {
      alert("✅ Payment held in escrow – both parties must confirm completion.");
      // Save completed job to DB
      if (db.jobs.length) db.jobs[db.jobs.length - 1].status = "Completed";
      saveDB();
      switchTab(3); // go to My Jobs
    }

    function renderPriceMatrix() {
      const container = document.getElementById('price-matrix');
      container.innerHTML = '';
      Object.keys(db.landscaperPrices[selectedLandscaper] || {}).forEach(service => {
        const prices = db.landscaperPrices[selectedLandscaper][service];
        let html = `<div class="bg-[#111111] rounded-3xl p-6 mb-6"><p class="font-semibold text-white mb-4">${service}</p><div class="grid grid-cols-3 gap-4">`;
        html += `<div class="text-center"><p class="text-xs text-white/70">Small</p><div class="bg-[#222222] rounded-2xl px-4 py-3 text-center mt-2">$${prices.Small}</div></div>`;
        html += `<div class="text-center"><p class="text-xs text-white/70">Medium</p><div class="bg-[#222222] rounded-2xl px-4 py-3 text-center mt-2">$${prices.Medium}</div></div>`;
        html += `<div class="text-center"><p class="text-xs text-white/70">Large</p><div class="bg-[#222222] rounded-2xl px-4 py-3 text-center mt-2">$${prices.Large}</div></div>`;
        html += `</div></div>`;
        container.innerHTML += html;
      });
    }

    function renderJobStatusTracking() {
      const container = document.getElementById('job-status-list');
      container.innerHTML = db.jobs.map(job => `
        <div class="card rounded-3xl p-6">
          <div class="flex justify-between items-center">
            <p class="font-semibold text-white">${job.service} – ${job.customer}</p>
            <span class="status-pill ${job.status === 'Completed' ? 'bg-[#34FF7A] text-black' : 'bg-[#FF9500] text-white'}">${job.status}</span>
          </div>
          <p class="text-xs text-white/70 mt-1">${job.address} • ${job.yardSize} yard</p>
          <p class="text-xs text-white/70">$${job.price}</p>
        </div>`).join('');
    }

        document.getElementById('service-selection-screen').classList.remove('hidden');

        switchTab(0);
      }
    }

      document.querySelectorAll('#screen > div').forEach(d => d.classList.add('hidden'));

      document.querySelectorAll('.nav-item').forEach((item,i) => i === tab ? item.classList.add('active','text-[#34FF7A]') : item.classList.remove('active','text-[#34FF7A]'));
    }
