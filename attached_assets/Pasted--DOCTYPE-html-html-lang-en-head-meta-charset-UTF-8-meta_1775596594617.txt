  <script>
    let db = { 
      users: {}, 
      jobs: [], 
      requests: [], 
      notifications: [], 
      reviews: [],
      servicePhotos: [],
      landscaperPrices: {} 
    };
    let currentUserType = null;
    let currentUser = null;
    let currentRequestFiles = [];
    let currentChatJobId = null;

    function loadDB() { const saved = localStorage.getItem('thelawn_db'); if(saved) db = JSON.parse(saved); }
    function saveDB() { localStorage.setItem('thelawn_db', JSON.stringify(db)); }

    /* ==================== CUSTOMER CLICKS LANDSCAPER → OPEN PROFILE ==================== */
    function viewLandscaperProfile(businessName) {
      if (currentUserType !== 'customer') {
        alert("❌ Only customers can view landscaper profiles.");
        return;
      }
      document.querySelectorAll('.iphone > div[id$="-screen"]').forEach(d => d.classList.add('hidden'));
      document.getElementById('landscaper-profile-screen').classList.remove('hidden');

      document.getElementById('landscaper-name').textContent = businessName || "GreenScape Pros";
      document.getElementById('landscaper-avatar-img').src = "https://picsum.photos/id/1015/300/300";
      document.getElementById('landscaper-established').textContent = "Est. 2018 • Sarasota, FL";
      document.getElementById('landscaper-about').textContent = "Professional landscaping services with over 10 years of experience.";

      // Hide edit buttons for customers
      document.getElementById('edit-avatar-hint').classList.add('hidden');
      document.getElementById('edit-background-btn').classList.add('hidden');
    }

    /* ==================== HOME SCREEN RECOMMENDED LANDSCAPERS – CLICKABLE TO PROFILE ==================== */
    function updateHomeScreen() {
      document.getElementById('home-greeting-name').textContent = currentUserType === 'customer' ? currentUser.first : currentUser.businessName;

      if (currentUserType === 'customer') {
        document.getElementById('customer-home-recommendations').classList.remove('hidden');
        document.getElementById('landscaper-home-recommendations').classList.add('hidden');
        document.getElementById('customer-nearby-pros').innerHTML = `
          <div onclick="viewLandscaperProfile('GreenScape Pros')" class="card rounded-3xl p-5 min-w-[220px] cursor-pointer">
            <div class="flex items-center gap-3">
              <div class="w-10 h-10 bg-[#34FF7A] rounded-2xl flex items-center justify-center text-2xl">🌿</div>
              <div>
                <p class="font-semibold">GreenScape Pros</p>
                <p class="text-xs text-white/70">2.4 mi away • 4.9 ★</p>
              </div>
            </div>
          </div>`;
      } else {
        document.getElementById('customer-home-recommendations').classList.add('hidden');
        document.getElementById('landscaper-home-recommendations').classList.remove('hidden');
        const nearbyRequests = db.requests.filter(r => true);
        let html = nearbyRequests.length ? nearbyRequests.map(r => `
          <div class="card rounded-3xl p-5">
            <p class="font-medium">${r.customer} • ${r.service}</p>
            <p class="text-xs text-white/70">${r.date} at ${r.time}</p>
            <button onclick="bookRequest(${r.id});event.stopImmediatePropagation()" class="mt-4 w-full bg-[#34FF7A] text-black py-3 rounded-3xl text-sm font-semibold">Accept Request</button>
          </div>`).join('') : `<p class="text-white/50 text-center py-8">No pending requests yet.</p>`;
        document.getElementById('landscaper-nearby-requests').innerHTML = html;
      }
    }

    /* ==================== LANDSCAPER PRIVACY SETTINGS ==================== */
    function openPrivacySettings() {
      if (currentUserType !== 'landscaper') return;
      document.getElementById('privacy-modal').classList.remove('hidden');
    }

    function savePrivacySettings() {
      document.getElementById('privacy-modal').classList.add('hidden');
      alert("✅ Privacy settings saved");
    }

    function togglePrivacy(cb) {
      console.log(cb.id + " toggled:", cb.checked);
    }

    /* ==================== OPTIMIZED PROFILE ACCESS LOGIC ==================== */
    function viewProfile() {
      toggleProfileDropdown();
      if (currentUserType === 'landscaper') {
        alert("❌ Landscapers cannot view other landscaper profiles.");
        return;
      }
      // Customer profile view is handled via viewLandscaperProfile from home/search
      alert("Please click a landscaper card from Home or Search to view a profile.");
    }

    /* ==================== BOOK FROM PROFILE ==================== */
    function startBookingFromProfile() {
      document.getElementById('landscaper-profile-screen').classList.add('hidden');
      document.getElementById('availability-screen').classList.remove('hidden');
      renderServiceTypeButtons();
    }

    /* ==================== APPOINTMENTS WITH CLIENT ADDRESS + MAPS (Landscaper) ==================== */
    function updateAppointmentsScreen() {
      const container = document.getElementById('appointments-list');
      container.innerHTML = '<p class="text-xs text-white/70 mb-3">Upcoming Appointments</p>';

      const futureJobs = db.jobs.filter(j => 
        j.status === "Future" && currentUserType === 'landscaper' && j.landscaper === currentUser.businessName
      );

      if (futureJobs.length === 0) {
        container.innerHTML += `<p class="text-white/50 text-center py-8">No upcoming appointments</p>`;
        return;
      }

      futureJobs.forEach(job => {
        const div = document.createElement('div');
        div.className = `card rounded-3xl p-5`;
        div.innerHTML = `
          <p class="font-medium">${job.service}</p>
          <p class="text-xs text-white/70">${job.date} at ${job.time}</p>
          <p onclick="openMaps('${job.address}')" class="text-[#34FF7A] underline cursor-pointer text-sm mt-3">${job.address}</p>
        `;
        container.appendChild(div);
      });
    }

    function openMaps(address) {
      const encoded = encodeURIComponent(address);
      if (navigator.userAgent.match(/iPhone|iPad/i)) {
        window.open(`https://maps.apple.com/?daddr=${encoded}`, '_blank');
      } else {
        window.open(`https://www.google.com/maps/dir/?api=1&destination=${encoded}`, '_blank');
      }
    }

    window.onload = () => {
      loadDB();
      console.log("%c✅ The Lawn – Customer clicks on searched/recommended landscaper → opens profile. BOOK button starts booking. Landscaper privacy settings added. Profile access logic optimized. Future appointments show client address with clickable maps 🌱", "color:#34FF7A;font-weight:bold");
    };
