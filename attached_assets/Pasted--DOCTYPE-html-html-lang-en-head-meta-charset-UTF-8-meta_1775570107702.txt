  <script>
    let db = { 
      users: {}, 
      jobs: [], 
      requests: [], 
      landscaperPrices: { "John Rivera Landscaping": { "Lawn Mowing": {Small:45,Medium:65,Large:120}, "Hedge Trimming":{Small:55,Medium:75,Large:95}, "Mulching":{Small:110,Medium:140,Large:180}, "Clean Up":{Small:30,Medium:40,Large:55} } } 
    };
    let currentUserType = null;
    let currentUser = null;
    let currentJobWatcher = null; // Hidden GPS watcher

    function loadDB() { const saved = localStorage.getItem('thelawn_db'); if(saved) db = JSON.parse(saved); }
    function saveDB() { localStorage.setItem('thelawn_db', JSON.stringify(db)); }

    function updateProfileIcon() {
      const icon = document.getElementById('profile-icon');
      if (!currentUser) return;
      if (currentUser.profilePicture) {
        icon.innerHTML = `<img src="${currentUser.profilePicture}" class="w-full h-full object-cover">`;
      } else {
        let letter = currentUserType === 'customer' && currentUser.first ? currentUser.first[0].toUpperCase() : currentUser.businessName ? currentUser.businessName[0].toUpperCase() : '';
        icon.innerHTML = letter ? `<span class="text-xl font-bold">${letter}</span>` : '👤';
      }
    }

    function updateBottomNav() {
      const dynamicTab = document.getElementById('dynamic-nav-tab');
      const label = document.getElementById('dynamic-nav-label');
      if (currentUserType === 'landscaper') {
        label.textContent = "Service Requests";
        dynamicTab.onclick = () => switchTab(1);
      } else {
        label.textContent = "Search";
        dynamicTab.onclick = () => switchTab(1);
      }
    }

    function startHiddenGPSWatcher(job) {
      if (currentJobWatcher) navigator.geolocation.clearWatch(currentJobWatcher);
      currentJobWatcher = navigator.geolocation.watchPosition(pos => {
        const serviceLat = job.serviceLat || 28.55; // demo coordinates
        const serviceLng = job.serviceLng || -81.35;
        const distance = calculateDistance(pos.coords.latitude, pos.coords.longitude, serviceLat, serviceLng);
        if (distance < 1 && job.status === "Future") {
          job.status = "In Progress";
          saveDB();
          updateMyJobsScreen();
          console.log("Hidden GPS: Job status changed to In Progress (within 1 mile)");
        }
      }, () => {}, { enableHighAccuracy: true, maximumAge: 0 });
    }

    function calculateDistance(lat1, lon1, lat2, lon2) {
      const R = 3958.8; // miles
      const dLat = (lat2 - lat1) * Math.PI / 180;
      const dLon = (lon2 - lon1) * Math.PI / 180;
      const a = Math.sin(dLat/2) * Math.sin(dLat/2) + Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon/2) * Math.sin(dLon/2);
      const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
      return R * c;
    }

    function completeCustomerRegistration() {
      const first = document.getElementById('reg-firstname').value.trim();
      const last = document.getElementById('reg-lastname').value.trim();
      const email = document.getElementById('reg-email').value.trim();
      const pass = document.getElementById('reg-password').value.trim();
      const phone = document.getElementById('reg-phone').value.trim();
      const address = document.getElementById('reg-address').value.trim();
      const zip = document.getElementById('reg-zip').value.trim();

      if (!first || !last || !email || !pass || !phone || !address || !zip) {
        alert("Please fill all fields including ZIP code");
        return;
      }

      currentUser = { first, last, email, password: pass, phone, address, zipcode: zip, type: 'customer', profilePicture: null };
      currentUserType = 'customer';
      document.getElementById('customer-register-screen').classList.add('hidden');
      document.getElementById('home-screen').classList.remove('hidden');
      updateBottomNav();
      updateHomeScreen();
      updateProfileIcon();
    }

    function completeLandscaperRegistration() {
      const businessName = document.getElementById('lreg-business-name').value.trim();
      const email = document.getElementById('lreg-email').value.trim();
      const pass = document.getElementById('lreg-password').value.trim();
      const phone = document.getElementById('lreg-phone').value.trim();
      const state = document.getElementById('lreg-state').value.trim();
      const zip = document.getElementById('lreg-zip').value.trim();
      const paymentPref = document.getElementById('lreg-payment-pref').value;

      if (!businessName || !email || !pass || !phone || !state || !zip || !paymentPref) {
        alert("Please fill all required fields");
        return;
      }

      currentUser = { businessName, email, password: pass, phone, state, zipcode: zip, paymentPreference: paymentPref, type: 'landscaper', profilePicture: null };
      currentUserType = 'landscaper';
      document.getElementById('landscaper-register-screen').classList.add('hidden');
      document.getElementById('home-screen').classList.remove('hidden');
      updateBottomNav();
      updateHomeScreen();
      updateProfileIcon();
    }

    function updateHomeScreen() {
      document.getElementById('home-greeting-name').textContent = currentUserType === 'customer' ? currentUser.first : currentUser.businessName;
      if (currentUserType === 'customer') {
        document.getElementById('customer-home-recommendations').classList.remove('hidden');
        document.getElementById('landscaper-home-recommendations').classList.add('hidden');
        document.getElementById('customer-zip-display').textContent = currentUser.zipcode || 'Unknown';
        document.getElementById('customer-nearby-pros').innerHTML = `
          <div class="card rounded-3xl p-5 min-w-[220px]">
            <div class="flex items-center gap-3">
              <div class="w-10 h-10 bg-[#34FF7A] rounded-2xl flex items-center justify-center text-2xl">🌿</div>
              <div>
                <p class="font-semibold">John Rivera Landscaping</p>
                <p class="text-xs text-white/70">2.4 mi away • 4.9 ★</p>
              </div>
            </div>
          </div>`;
      } else {
        document.getElementById('customer-home-recommendations').classList.add('hidden');
        document.getElementById('landscaper-home-recommendations').classList.remove('hidden');
        document.getElementById('landscaper-zip-display').textContent = currentUser.zipcode || 'Unknown';
        const nearbyRequests = db.requests.filter(r => true);
        let html = '';
        if (nearbyRequests.length) {
          html = nearbyRequests.map(r => `
            <div class="card rounded-3xl p-5">
              <p class="font-medium">${r.customer} • ${r.service}</p>
              <p class="text-xs text-white/70">${r.date} at ${r.time}</p>
              <button onclick="bookRequest(${r.id})" class="mt-4 w-full bg-[#34FF7A] text-black py-3 rounded-3xl text-sm font-semibold">Book This Request</button>
            </div>`).join('');
        } else {
          html = `<p class="text-white/50 text-center py-8">No pending requests in your area yet.</p>`;
        }
        document.getElementById('landscaper-nearby-requests').innerHTML = html;
      }
    }

    function bookRequest(id) {
      const req = db.requests.find(r => r.id === id);
      if (!req) return;
      req.status = "Booked";
      db.jobs.push({
        id: Date.now(),
        customer: req.customer,
        service: req.service,
        date: req.date,
        time: req.time,
        status: "Future",
        address: req.address,
        serviceLat: 28.55, // demo coordinates for GPS check
        serviceLng: -81.35,
        type: "landscaper"
      });
      saveDB();
      alert("✅ Request booked! Added to Future Services.");
      updateHomeScreen();
      switchTab(3);
      if (currentUserType === 'landscaper') startHiddenGPSWatcher(db.jobs[db.jobs.length - 1]);
    }

    function updateMyJobsScreen() {
      const container = document.getElementById('job-status-list');
      const activeJobs = db.jobs.filter(j => j.type === 'landscaper' && (j.status === "Future" || j.status === "In Progress"));
      let html = '';
      if (activeJobs.length) {
        html = activeJobs.map(job => `
          <div class="card rounded-3xl p-6">
            <p class="font-semibold">${job.service} – ${job.customer}</p>
            <p class="text-xs text-white/70">${job.date} at ${job.time} • ${job.status}</p>
            ${job.status === "In Progress" ? `<button onclick="markJobComplete(${job.id})" class="mt-4 w-full bg-[#34FF7A] text-black py-3 rounded-3xl text-sm font-semibold">Mark as Complete</button>` : ''}
          </div>`).join('');
      } else {
        html = `<p class="text-white/50 text-center py-8">No active jobs at the moment.</p>`;
      }
      container.innerHTML = html;
    }

    function markJobComplete(id) {
      const job = db.jobs.find(j => j.id === id);
      if (job) job.status = "Completed";
      saveDB();
      updateMyJobsScreen();
      alert("✅ Job marked as completed.");
    }

    function updateAppointmentsScreen() {
      const container = document.getElementById('scheduled-appointments-list');
      const scheduled = db.jobs.filter(j => j.type === 'landscaper' && j.status === "Future");
      let html = '';
      if (scheduled.length) {
        html = scheduled.map(job => `
          <div class="card rounded-3xl p-6">
            <p class="font-semibold">${job.service} – ${job.customer}</p>
            <p class="text-xs text-white/70">${job.date} at ${job.time}</p>
          </div>`).join('');
      } else {
        html = `<p class="text-white/50 text-center py-8">No scheduled appointments yet.</p>`;
      }
      container.innerHTML = html;
    }

    function switchTab(tab) {
      document.querySelectorAll('.iphone > div[id$="-screen"]').forEach(d => d.classList.add('hidden'));
      const screens = ['home-screen','service-requests-screen','appointments-screen','myjobs-screen','profile-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');
      if (tab === 0) updateHomeScreen();
      if (tab === 1 && currentUserType === 'landscaper') updateHomeScreen();
      if (tab === 2 && currentUserType === 'landscaper') updateAppointmentsScreen();
      if (tab === 3 && currentUserType === 'landscaper') updateMyJobsScreen();
    }

    window.onload = () => {
      loadDB();
      console.log("%c✅ The Lawn – Hidden GPS for auto In Progress + simplified landscaper tabs + all tracking removed 🌱", "color:#34FF7A;font-weight:bold");
      updateBottomNav();
    };
