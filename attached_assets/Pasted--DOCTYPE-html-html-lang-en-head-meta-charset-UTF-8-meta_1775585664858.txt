    /* ==================== ADVANCED IMAGE COMPRESSION + PROGRESSIVE LOADING (unchanged) ==================== */
    async function compressImage(file) {
      return new Promise(async (resolve) => {
        const reader = new FileReader();
        reader.onload = (e) => {
          const img = new Image();
          img.onload = () => {
            let { width, height } = img;
            const MAX_DIM = 1400;
            if (width > MAX_DIM || height > MAX_DIM) {
              const ratio = MAX_DIM / Math.max(width, height);
              width = Math.floor(width * ratio);
              height = Math.floor(height * ratio);
            }
            const canvas = document.createElement('canvas');
            canvas.width = width;
            canvas.height = height;
            const ctx = canvas.getContext('2d');
            ctx.drawImage(img, 0, 0, width, height);

            if (canvas.toDataURL('image/webp').indexOf('data:image/webp') === 0) {
              let quality = 0.94;
              const tryWebP = () => {
                canvas.toBlob((blob) => {
                  if (blob.size > 900 * 1024 && quality > 0.65) {
                    quality -= 0.04;
                    tryWebP();
                  } else {
                    resolve(new File([blob], file.name.replace(/\.\w+$/, '.webp'), { type: 'image/webp', lastModified: Date.now() }));
                  }
                }, 'image/webp', quality);
              };
              tryWebP();
            } else {
              let quality = 0.92;
              const tryJPEG = () => {
                canvas.toBlob((blob) => {
                  if (blob.size > 900 * 1024 && quality > 0.60) {
                    quality -= 0.05;
                    tryJPEG();
                  } else {
                    resolve(new File([blob], file.name.replace(/\.\w+$/, '.jpg'), { type: 'image/jpeg', lastModified: Date.now() }));
                  }
                }, 'image/jpeg', quality);
              };
              tryJPEG();
            }
          };
          img.src = e.target.result;
        };
        reader.readAsDataURL(file);
      });
    }

    async function handleRequestFileUpload(e) {
      const files = Array.from(e.target.files).slice(0, 5);
      const previewContainer = document.getElementById('request-file-preview');
      previewContainer.innerHTML = '';
      currentRequestFiles = [];

      for (let i = 0; i < files.length; i++) {
        const file = files[i];
        if (!file.type.startsWith('image/')) continue;

        const thumb = document.createElement('div');
        thumb.className = `preview-thumb w-20 h-20 relative preview-stagger`;
        thumb.style.animationDelay = `${i * 60}ms`;
        thumb.innerHTML = `<div class="preview-skeleton w-full h-full">OPTIMIZING</div>`;
        previewContainer.appendChild(thumb);

        const compressedFile = await compressImage(file);
        currentRequestFiles.push(compressedFile);

        const reader = new FileReader();
        reader.onload = (ev) => {
          thumb.innerHTML = `
            <img src="${ev.target.result}" alt="preview">
            <button onclick="removePreviewFile(this)" class="remove-btn absolute -top-1 -right-1 w-6 h-6 bg-red-500 text-white rounded-full flex items-center justify-center text-xs leading-none shadow-md">×</button>
            <div class="compression-badge">COMPRESSED</div>
          `;
          const imgEl = thumb.querySelector('img');
          setTimeout(() => imgEl.classList.add('loaded'), 120);
        };
        reader.readAsDataURL(compressedFile);
      }
    }

    function removePreviewFile(btn) {
      const thumb = btn.parentElement;
      const index = Array.from(thumb.parentNode.children).indexOf(thumb);
      currentRequestFiles.splice(index, 1);
      thumb.remove();
    }

        document.getElementById('landscaper-register-screen').classList.remove('hidden');
      }
    }

    function updateHomeScreen() {
      document.getElementById('home-greeting-name').textContent = currentUserType === 'customer' ? currentUser.first : currentUser.businessName;
      if (currentUserType === 'customer') {
        document.getElementById('customer-home-recommendations').classList.remove('hidden');
        document.getElementById('landscaper-home-recommendations').classList.add('hidden');
      } else {
        document.getElementById('customer-home-recommendations').classList.add('hidden');
        document.getElementById('landscaper-home-recommendations').classList.remove('hidden');
      }
    }

    function updateProfileIcon() {
      const icon = document.getElementById('profile-icon');
      if (!currentUser) return;
      if (currentUser.profilePicture) {
        icon.innerHTML = `<img src="${currentUser.profilePicture}" class="w-full h-full object-cover">`;
      } else {
        let letter = currentUserType === 'customer' && currentUser.first ? currentUser.first[0].toUpperCase() : currentUser.businessName ? currentUser.businessName[0].toUpperCase() : '';
        icon.innerHTML = letter ? `<span class="text-xl font-bold">${letter}</span>` : '👤';
      }
    }
