  <script>
    let db = { users: {}, jobs: [], requests: [], landscaperPrices: { "John Rivera Landscaping": { "Lawn Mowing": {Small:45,Medium:65,Large:120}, "Hedge Trimming":{Small:55,Medium:75,Large:95}, "Mulching":{Small:110,Medium:140,Large:180}, "Clean Up":{Small:30,Medium:40,Large:55} } } };
    let currentUserType = null;
    let currentUser = null;

    function loadDB() { const saved = localStorage.getItem('thelawn_db'); if(saved) db = JSON.parse(saved); }
    function saveDB() { localStorage.setItem('thelawn_db', JSON.stringify(db)); }

    function updateProfileIcon() {
      const icon = document.getElementById('profile-icon');
      if (!currentUser) return;
      if (currentUser.profilePicture) {
        icon.innerHTML = `<img src="${currentUser.profilePicture}" class="w-full h-full object-cover">`;
      } else {
        let letter = currentUserType === 'customer' && currentUser.first ? currentUser.first[0].toUpperCase() : currentUser.businessName ? currentUser.businessName[0].toUpperCase() : '';
        icon.innerHTML = letter ? `<span class="text-xl font-bold">${letter}</span>` : '👤';
      }
    }

    function updateBottomNav() {
      const requestsTab = document.getElementById('dynamic-requests-tab');
      const label = document.getElementById('dynamic-requests-label');
      if (currentUserType === 'customer') {
        label.textContent = "Requests";
        requestsTab.onclick = () => switchTab(5);
      } else {
        label.textContent = "Service Requests";
        requestsTab.onclick = () => switchTab(1);
      }
    }

    function viewProfile() {
      toggleProfileDropdown();
      switchTab(4);
    }

    function completeCustomerRegistration() {
      const first = document.getElementById('reg-firstname').value.trim();
      const last = document.getElementById('reg-lastname').value.trim();
      const email = document.getElementById('reg-email').value.trim();
      const pass = document.getElementById('reg-password').value.trim();
      const phone = document.getElementById('reg-phone').value.trim();
      const address = document.getElementById('reg-address').value.trim();
      const zip = document.getElementById('reg-zip').value.trim();

      if (!first || !last || !email || !pass || !phone || !address || !zip) {
        alert("Please fill all fields including ZIP code");
        return;
      }

      currentUser = { first, last, email, password: pass, phone, address, zipcode: zip, type: 'customer', profilePicture: null };
      currentUserType = 'customer';
      document.getElementById('customer-register-screen').classList.add('hidden');
      document.getElementById('home-screen').classList.remove('hidden');
      updateBottomNav();
      updateHomeScreen();
      updateProfileIcon();
    }

    function completeLandscaperRegistration() {
      const businessName = document.getElementById('lreg-business-name').value.trim();
      const email = document.getElementById('lreg-email').value.trim();
      const pass = document.getElementById('lreg-password').value.trim();
      const phone = document.getElementById('lreg-phone').value.trim();
      const state = document.getElementById('lreg-state').value.trim();
      const zip = document.getElementById('lreg-zip').value.trim();
      const paymentPref = document.getElementById('lreg-payment-pref').value;

      if (!businessName || !email || !pass || !phone || !state || !zip || !paymentPref) {
        alert("Please fill all required fields");
        return;
      }

      currentUser = { businessName, email, password: pass, phone, state, zipcode: zip, paymentPreference: paymentPref, type: 'landscaper', profilePicture: null };
      currentUserType = 'landscaper';
      document.getElementById('landscaper-register-screen').classList.add('hidden');
      document.getElementById('home-screen').classList.remove('hidden');
      updateBottomNav();
      updateHomeScreen();
      updateProfileIcon();
    }

    function updateHomeScreen() {
      document.getElementById('home-greeting-name').textContent = currentUserType === 'customer' ? currentUser.first : currentUser.businessName;
      if (currentUserType === 'customer') {
        document.getElementById('customer-home-recommendations').classList.remove('hidden');
        document.getElementById('landscaper-home-recommendations').classList.add('hidden');
        document.getElementById('customer-zip-display').textContent = currentUser.zipcode || 'Unknown';
        document.getElementById('customer-nearby-pros').innerHTML = `<div class="card rounded-3xl p-5 min-w-[220px]"><div class="flex items-center gap-3"><div class="w-10 h-10 bg-[#34FF7A] rounded-2xl flex items-center justify-center text-2xl">🌿</div><div><p class="font-semibold">John Rivera Landscaping</p><p class="text-xs text-white/70">2.4 mi away • 4.9 ★</p></div></div></div>`;
      } else {
        document.getElementById('customer-home-recommendations').classList.add('hidden');
        document.getElementById('landscaper-home-recommendations').classList.remove('hidden');
        document.getElementById('landscaper-zip-display').textContent = currentUser.zipcode || 'Unknown';
        // Pending requests demo
        const nearbyRequests = db.requests.filter(r => true);
        let html = nearbyRequests.length ? nearbyRequests.map(r => `<div class="card rounded-3xl p-5"><p class="font-medium">${r.customer} • ${r.service}</p><p class="text-xs text-white/70">${r.date} at ${r.time}</p><button onclick="bookRequest(${r.id})" class="mt-4 w-full bg-[#34FF7A] text-black py-3 rounded-3xl text-sm font-semibold">Book This Request</button></div>`).join('') : `<p class="text-white/50 text-center py-8">No pending requests yet.</p>`;
        document.getElementById('landscaper-nearby-requests').innerHTML = html;
      }
    }

    function bookRequest(id) {
      const req = db.requests.find(r => r.id === id);
      if (!req) return;
      req.status = "Accepted";
      req.offeredPrice = 85; // demo price
      db.jobs.push({ id: Date.now(), customer: req.customer, service: req.service, date: req.date, time: req.time, status: "Future", address: req.address, type: "landscaper" });
      saveDB();
      alert("✅ Request accepted and booked!");
      updateHomeScreen();
      switchTab(3);
    }

    function updateMyRequestsScreen() {
      const container = document.getElementById('customer-requests-list');
      const customerRequests = db.requests.filter(r => r.customer === (currentUser.first + " " + currentUser.last));
      let html = '';
      if (customerRequests.length) {
        html = customerRequests.map(r => {
          if (r.status === "Accepted") {
            return `<div class="card rounded-3xl p-5"><p class="font-medium">${r.service}</p><p class="text-xs text-white/70">${r.date} at ${r.time}</p><p class="text-[#34FF7A] mt-2">Offer received: $${r.offeredPrice}</p><div class="flex gap-3 mt-4"><button onclick="acceptOffer(${r.id})" class="flex-1 bg-[#34FF7A] text-black py-3 rounded-3xl text-sm font-semibold">Accept Offer</button><button onclick="declineOffer(${r.id})" class="flex-1 bg-[#222222] py-3 rounded-3xl text-sm font-semibold">Decline</button></div></div>`;
          }
          return `<div class="card rounded-3xl p-5"><p class="font-medium">${r.service}</p><p class="text-xs text-white/70">${r.date} at ${r.time}</p><p class="text-white/50">Pending landscaper response...</p></div>`;
        }).join('');
      } else {
        html = `<p class="text-white/50 text-center py-8">No requests yet.</p>`;
      }
      container.innerHTML = html;
    }

    function acceptOffer(id) {
      const req = db.requests.find(r => r.id === id);
      if (!req) return;
      req.status = "Booked";
      saveDB();
      alert("✅ Offer accepted – proceeding to payment");
      // Trigger normal payment process
      document.getElementById('my-requests-screen').classList.add('hidden');
      document.getElementById('pay-screen').classList.remove('hidden');
      // Demo price
      document.getElementById('service-name-display').textContent = req.service;
      document.getElementById('subtotal-display').textContent = `$${req.offeredPrice}.00`;
      document.getElementById('total-display').textContent = `$${req.offeredPrice}.00`;
      document.getElementById('final-total').textContent = `$${req.offeredPrice}.00`;
    }

    function declineOffer(id) {
      const req = db.requests.find(r => r.id === id);
      if (req) req.status = "Declined";
      saveDB();
      updateMyRequestsScreen();
      alert("Request declined");
    }

    function switchTab(tab) {
      document.querySelectorAll('.iphone > div[id$="-screen"]').forEach(d => d.classList.add('hidden'));
      const screens = ['home-screen','service-requests-screen','appointments-screen','myjobs-screen','profile-screen','my-requests-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');
      if (tab === 0) updateHomeScreen();
      if (tab === 5 && currentUserType === 'customer') updateMyRequestsScreen();
    }

    window.onload = () => {
      loadDB();
      console.log("%c✅ The Lawn – Customer bottom nav updated + View Profile in dropdown + Offer notification & payment flow complete 🌱", "color:#34FF7A;font-weight:bold");
    };
  </script>
