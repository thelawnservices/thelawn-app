  <script>
    let isLandscaperMode = true;

    function toggleProfileMode() {
      isLandscaperMode = !isLandscaperMode;
      document.getElementById('landscaper-profile').classList.toggle('hidden', !isLandscaperMode);
      document.getElementById('customer-profile').classList.toggle('hidden', isLandscaperMode);
      document.getElementById('profile-mode-label').textContent = isLandscaperMode ? "Landscaper View" : "Customer View";
    }

    function savePaymentMethod() {
      const select = document.getElementById('payment-select');
      const msg = document.getElementById('payment-saved-msg');
      msg.classList.remove('hidden');
      setTimeout(() => msg.classList.add('hidden'), 2500);
      alert("✅ Preferred payment method saved: " + select.value);
    }

    function switchTab(tab) {
      document.querySelectorAll('#screen > div').forEach(d => d.classList.add('hidden'));
      const screens = ['home-screen','search-screen','appointments-screen','myjobs-screen','profile-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');
      document.querySelectorAll('.nav-item').forEach((item,i) => i === tab ? item.classList.add('active','text-[#34FF7A]') : item.classList.remove('active','text-[#34FF7A]'));
    }

    function startBooking() { alert("Starting booking flow..."); }
    function toggleNotifications() { alert("🛎️ Notifications panel would open here."); }

    };
