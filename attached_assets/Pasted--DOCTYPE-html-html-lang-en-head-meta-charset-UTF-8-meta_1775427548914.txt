  <script>
    let currentUserType = null;
    let customerYardSize = null;
    let currentLocation = null;

    function loginDemo(type) {
      currentUserType = type;
      document.getElementById('auth-screen').classList.add('hidden');
      document.getElementById('app-container').classList.remove('hidden');
      switchTab(0);
    }

    function showRegistration(type) {
      if (type === 'customer') {
        document.getElementById('auth-screen').classList.add('hidden');
        document.getElementById('customer-register-screen').classList.remove('hidden');
      }
    }

    function completeCustomerRegistration() {
      const email = document.getElementById('reg-email').value;
      const phone = document.getElementById('reg-phone').value;
      const dob = document.getElementById('reg-dob').value;
      const address = document.getElementById('reg-address').value;
      customerYardSize = document.getElementById('reg-yard-size').value;

      if (!email || !phone || !dob || !address || !customerYardSize) {
        alert("Please fill all fields");
        return;
      }

      currentUserType = 'customer';
      document.getElementById('customer-register-screen').classList.add('hidden');
      document.getElementById('app-container').classList.remove('hidden');
      switchTab(0);
    }

    function toggleProfileMode() {
      const isLandscaper = document.getElementById('profile-mode-label').textContent.includes('Landscaper');
      document.getElementById('landscaper-profile').classList.toggle('hidden', !isLandscaper);
      document.getElementById('customer-profile').classList.toggle('hidden', isLandscaper);
      document.getElementById('profile-mode-label').textContent = isLandscaper ? "Customer View" : "Landscaper View";
    }

    function saveLandscaperPrices() {
      alert("✅ Service prices updated based on yard size");
    }

    function getCurrentLocation() {
      if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(pos => {
          currentLocation = { lat: pos.coords.latitude, lng: pos.coords.longitude };
          alert("📍 Location acquired: " + currentLocation.lat + ", " + currentLocation.lng);
        });
      } else {
        alert("Geolocation not supported");
      }
    }

    function renderHorizontalTrusted() {
      const container = document.getElementById('trusted-horizontal');
      container.innerHTML = '';
      const topLandscapers = [
        { name: "John Rivera Landscaping", rating: 4.9, jobs: 142 },
        { name: "Sarah’s Lawn Care", rating: 5.0, jobs: 98 },
        { name: "GreenScape Pros", rating: 4.8, jobs: 87 },
        { name: "Elite Lawn Services", rating: 4.9, jobs: 65 },
        { name: "FreshCut Landscaping", rating: 5.0, jobs: 112 },
        { name: "Premier Turf Care", rating: 4.7, jobs: 79 },
        { name: "Nature’s Edge Lawn", rating: 4.9, jobs: 53 }
      ];
      topLandscapers.forEach(l => {
        const html = `
          <div onclick="startBooking()" class="card flex-shrink-0 w-52 rounded-3xl p-5 cursor-pointer btn-hover micro-tap">
            <div class="text-5xl mb-3">🌿</div>
            <p class="font-semibold text-white">${l.name}</p>
            <p class="text-xs text-white">${l.rating} ★ • ${l.jobs} jobs</p>
          </div>`;
        container.innerHTML += html;
      });
    }

    function switchTab(tab) {
      document.querySelectorAll('#screen > div').forEach(d => d.classList.add('hidden'));
      const screens = ['home-screen','search-screen','appointments-screen','myjobs-screen','profile-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');
      document.querySelectorAll('.nav-item').forEach((item,i) => i === tab ? item.classList.add('active','text-[#34FF7A]') : item.classList.remove('active','text-[#34FF7A]'));
    }

    function startBooking() { switchTab(2); }
    function goBack() { switchTab(0); }
    function toggleNotifications() { alert("🛎️ Notifications"); }

    window.onload = () => {
      console.log("%c✅ COMPLETE The Lawn App – Geolocation + Yard Size Pricing + Visibility Rules + Full Registration + Appointments Section 🌱", "color:#34FF7A;font-weight:bold");
      renderHorizontalTrusted();
      switchTab(0);
    };
  </script>
