    /* ADVANCED IMAGE COMPRESSION (unchanged from previous) */
    async function compressImage(file) {
      return new Promise(async (resolve) => {
        const reader = new FileReader();
        reader.onload = (e) => {
          const img = new Image();
          img.onload = () => {
            let { width, height } = img;
            const MAX_DIM = 1400;
            if (width > MAX_DIM || height > MAX_DIM) {
              const ratio = MAX_DIM / Math.max(width, height);
              width = Math.floor(width * ratio);
              height = Math.floor(height * ratio);
            }
            const canvas = document.createElement('canvas');
            canvas.width = width;
            canvas.height = height;
            const ctx = canvas.getContext('2d');
            ctx.drawImage(img, 0, 0, width, height);

            if (typeof canvas.toDataURL === 'function' && canvas.toDataURL('image/webp').indexOf('data:image/webp') === 0) {
              let quality = 0.92;
              const tryWebP = () => {
                canvas.toBlob((blob) => {
                  if (blob.size > 1024 * 1024 && quality > 0.65) {
                    quality -= 0.04;
                    tryWebP();
                  } else {
                    resolve(new File([blob], file.name.replace(/\.\w+$/, '.webp'), { type: 'image/webp', lastModified: Date.now() }));
                  }
                }, 'image/webp', quality);
              };
              tryWebP();
            } else {
              let quality = 0.92;
              const tryJPEG = () => {
                canvas.toBlob((blob) => {
                  if (blob.size > 1024 * 1024 && quality > 0.60) {
                    quality -= 0.05;
                    tryJPEG();
                  } else {
                    resolve(new File([blob], file.name.replace(/\.\w+$/, '.jpg'), { type: 'image/jpeg', lastModified: Date.now() }));
                  }
                }, 'image/jpeg', quality);
              };
              tryJPEG();
            }
          };
          img.src = e.target.result;
        };
        reader.readAsDataURL(file);
      });
    }

    /* PROGRESSIVE IMAGE LOADING + PREVIEW */
    async function handleRequestFileUpload(e) {
      const files = Array.from(e.target.files).slice(0, 5);
      const previewContainer = document.getElementById('request-file-preview');
      previewContainer.innerHTML = '';

      currentRequestFiles = [];

      for (let file of files) {
        if (!file.type.startsWith('image/')) continue;

        // 1. INSTANT SKELETON + SHIMMER
        const thumb = document.createElement('div');
        thumb.className = "preview-thumb w-20 h-20 relative";
        thumb.innerHTML = `
          <div class="preview-skeleton w-full h-full">LOADING</div>
          <div class="absolute inset-0 flex items-center justify-center text-[10px] font-bold text-white/60">PROGRESSIVE</div>
        `;
        previewContainer.appendChild(thumb);

        // 2. Compress in background
        const compressedFile = await compressImage(file);
        currentRequestFiles.push(compressedFile);

        // 3. Create progressive preview: low-res first → full crisp image
        const reader = new FileReader();
        reader.onload = (ev) => {
          // Replace skeleton with actual image container
          thumb.innerHTML = `
            <img src="${ev.target.result}" alt="preview">
            <button onclick="removePreviewFile(this)" class="remove-btn absolute -top-1 -right-1 w-6 h-6 bg-red-500 text-white rounded-full flex items-center justify-center text-xs leading-none shadow-md">×</button>
            <div class="compression-badge">COMPRESSED</div>
          `;

          // Force browser to render low-res first, then fade in crisp
          const imgEl = thumb.querySelector('img');
          imgEl.style.opacity = '0';

          // Small delay to simulate progressive load
          setTimeout(() => {
            imgEl.classList.add('loaded');
          }, 80);
        };
        reader.readAsDataURL(compressedFile);
      }
    }

    function removePreviewFile(btn) {
      const thumb = btn.parentElement;
      const index = Array.from(thumb.parentNode.children).indexOf(thumb);
      currentRequestFiles.splice(index, 1);
      thumb.remove();
    }

    function startNewRequest() {
      const specificNeeds = document.getElementById('new-request-specific-needs').value.trim();
      if (!specificNeeds && currentRequestFiles.length === 0) {
        if (!confirm("No specific needs or photos added. Continue?")) return;
      }
      window.specificNeeds = specificNeeds || "No specific needs provided";
      window.attachedFiles = currentRequestFiles;
      document.getElementById('my-requests-screen').classList.add('hidden');
      document.getElementById('availability-screen').classList.remove('hidden');
      renderServiceTypeButtons();
      // Reset form
      document.getElementById('new-request-specific-needs').value = '';
      document.getElementById('request-file-preview').innerHTML = '';
      currentRequestFiles = [];
    }
