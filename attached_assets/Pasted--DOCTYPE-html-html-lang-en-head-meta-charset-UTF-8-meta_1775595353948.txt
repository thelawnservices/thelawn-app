  <script>
    let db = { 
      users: {}, 
      jobs: [], 
      requests: [], 
      notifications: [], 
      reviews: [],
      servicePhotos: [],
      landscaperPrices: {} 
    };
    let currentUserType = null;
    let currentUser = null;
    let currentRequestFiles = [];
    let currentChatJobId = null;

    function loadDB() { const saved = localStorage.getItem('thelawn_db'); if(saved) db = JSON.parse(saved); }
    function saveDB() { localStorage.setItem('thelawn_db', JSON.stringify(db)); }

    /* ==================== ACCEPT REQUEST → ADDS TO APPOINTMENTS ==================== */
    function bookRequest(id) {
      const req = db.requests.find(r => r.id === id);
      if (!req) return;

      // Mark request as accepted
      req.status = "Accepted";
      req.offeredPrice = 85;

      // Create Future appointment (this is what goes into Appointments section)
      const newJob = {
        id: Date.now(),
        customer: req.customer,
        service: req.service,
        date: req.date,
        time: req.time,
        status: "Future",           // ← This makes it appear in Appointments
        address: req.address,
        type: "landscaper",
        landscaper: currentUser.businessName
      };
      db.jobs.push(newJob);

      saveDB();

      alert("✅ Request accepted! It has been added to your Appointments section as a Future appointment.");

      // Refresh both screens
      updateAppointmentsScreen();
      updateMyJobsScreen();
      updateMyRequestsScreen();   // so customer sees the acceptance
      switchTab(2);               // jump to Appointments for landscaper
    }

    /* ==================== APPOINTMENTS SCREEN (Future jobs) ==================== */
    function updateAppointmentsScreen() {
      const container = document.getElementById('appointments-list');
      container.innerHTML = '<p class="text-xs text-white/70 mb-3">Upcoming Appointments</p>';

      const futureJobs = db.jobs.filter(j => 
        j.status === "Future" && 
        ((currentUserType === 'landscaper' && j.landscaper === currentUser.businessName) || 
         (currentUserType === 'customer' && j.customer === (currentUser.first + " " + currentUser.last)))
      );

      if (futureJobs.length === 0) {
        container.innerHTML += `<p class="text-white/50 text-center py-8">No upcoming appointments</p>`;
        return;
      }

      futureJobs.forEach(job => {
        const div = document.createElement('div');
        div.className = `card rounded-3xl p-5`;
        div.innerHTML = `<p class="font-medium">${job.service}</p><p class="text-xs text-white/70">${job.date} at ${job.time}</p>`;
        container.appendChild(div);
      });
    }

    /* ==================== MY JOBS SCREEN (Active + Completed) ==================== */
    function updateMyJobsScreen() {
      const activeContainer = document.getElementById('active-jobs-list');
      const completedContainer = document.getElementById('completed-jobs-list');
      activeContainer.innerHTML = '';
      completedContainer.innerHTML = '';

      const myJobs = db.jobs.filter(j => 
        (currentUserType === 'customer' && j.customer === (currentUser.first + " " + currentUser.last)) ||
        (currentUserType === 'landscaper' && j.landscaper === currentUser.businessName)
      );

      // Active only
      const activeJobs = myJobs.filter(j => ['In Progress', 'Active', 'Arrived'].includes(j.status));
      if (activeJobs.length === 0) {
        activeContainer.innerHTML = `<p class="text-white/50 text-center py-8">No active jobs right now</p>`;
      } else {
        activeJobs.forEach(job => {
          const div = document.createElement('div');
          div.className = `card rounded-3xl p-5`;
          div.innerHTML = `
            <div class="flex justify-between items-start">
              <div>
                <p class="font-semibold">${job.service}</p>
                <p class="text-xs text-white/70">${job.date} • ${job.time}</p>
                <p class="text-sm mt-2 text-[#34FF7A]">${job.status}</p>
              </div>
              <div class="flex gap-3">
                <button onclick="openChat(${job.id});event.stopImmediatePropagation()" class="w-9 h-9 bg-[#222222] rounded-2xl flex items-center justify-center text-xl">💬</button>
                <button onclick="simulateCall(${job.id});event.stopImmediatePropagation()" class="w-9 h-9 bg-[#34FF7A] text-black rounded-2xl flex items-center justify-center text-xl">📞</button>
              </div>
            </div>
          `;
          activeContainer.appendChild(div);
        });
      }

      // Completed
      const completedJobs = myJobs.filter(j => j.status === "Completed");
      if (completedJobs.length === 0) {
        completedContainer.innerHTML = `<p class="text-white/50 text-center py-8">No completed jobs yet</p>`;
      } else {
        completedJobs.forEach(job => {
          const div = document.createElement('div');
          div.className = `card rounded-3xl p-5 opacity-75`;
          div.innerHTML = `<p class="font-medium">${job.service}</p><p class="text-xs text-white/70">${job.date} • ${job.time}</p><p class="text-[#34FF7A] text-sm">✓ Completed</p>`;
          completedContainer.appendChild(div);
        });
      }
    }

    /* Rest of the app (progressive loading, passkey, chat, etc.) remains fully functional */
    function switchTab(tab) {
      document.querySelectorAll('.iphone > div[id$="-screen"]').forEach(d => d.classList.add('hidden'));
      const screens = ['home-screen','service-requests-screen','appointments-screen','myjobs-screen','landscaper-profile-screen','my-requests-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');
      if (tab === 2) updateAppointmentsScreen();
      if (tab === 3) updateMyJobsScreen();
      if (tab === 5 && currentUserType === 'customer') updateMyRequestsScreen();
    }

    window.onload = () => {
      loadDB();
      console.log("%c✅ The Lawn – When a landscaper accepts a customer request it now adds it to the Appointments section as a Future appointment 🌱", "color:#34FF7A;font-weight:bold");
    };
  </script>
