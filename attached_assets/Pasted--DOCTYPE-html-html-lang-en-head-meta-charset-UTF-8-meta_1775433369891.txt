      notificationsEnabled = !notificationsEnabled;

      updateNotificationBadge();

    function updateNotificationBadge() {
      const bell = document.querySelector('.header span[onclick="toggleNotifications()"]');
      if (!bell) return;
      if (notificationsEnabled) {
        bell.innerHTML = `🛎️`;
        bell.onclick = toggleNotifications;
      } else {
        bell.innerHTML = `🛎️<span class="absolute -top-1 -right-1 bg-[#FF3B30] text-white text-[10px] font-bold rounded-full h-4 w-4 flex items-center justify-center">${notificationCount}</span>`;
        bell.onclick = () => alert("Notifications are turned OFF");
      }
    }

    function pushNotification(title, message) {
      if (!notificationsEnabled) { notificationCount++; updateNotificationBadge(); return; }
      const list = document.getElementById('notification-list');
      const html = `<div class="refined-notif rounded-3xl p-4 flex gap-3"><div class="text-3xl">🛎️</div><div><p class="font-medium text-white">${title}</p><p class="text-xs text-white/70">${message}</p></div></div>`;
      list.innerHTML = html + list.innerHTML;
      document.getElementById('notification-panel').classList.remove('hidden');
    }

      if (!notificationsEnabled) return;

    function validatePayment() {
      const method = document.getElementById('payment-method').value;
      if (method === 'applepay') return true;
      if (method === 'debit') {
        const card = (document.getElementById('card-number')?.value || '').replace(/\s/g,'');
        const expiry = document.getElementById('expiry')?.value || '';
        const cvv = document.getElementById('cvv')?.value || '';
        if (!/^\d{16}$/.test(card)) return "Invalid card number";
        if (!/^\d{2}\/\d{2}$/.test(expiry)) return "Invalid expiry (MM/YY)";
        if (!/^\d{3,4}$/.test(cvv)) return "Invalid CVV";
        return true;
      }
      if (method === 'venmo') {
        const username = document.getElementById('venmo-username')?.value || '';
        if (!username || !username.startsWith('@')) return "Invalid Venmo username";
        return true;
      }
      if (method === 'paypal') {
        const email = document.getElementById('paypal-email')?.value || '';
        if (!email.includes('@')) return "Invalid PayPal email";
        return true;
      }
      if (method === 'cashapp') {
        const tag = document.getElementById('cashapp-tag')?.value || '';
        if (!tag || !tag.startsWith('$')) return "Invalid Cash App $tag";
        return true;
      }
      return true;
    }

    function selectTip(el) {
      document.querySelectorAll('#pay-screen button:not([onclick^="selectNoTip"])').forEach(b => b.classList.remove('bg-[#34FF7A]', 'text-black'));
      el.classList.add('bg-[#34FF7A]', 'text-black');
      const percent = parseFloat(el.textContent);
      const subtotal = parseFloat(document.getElementById('subtotal-display').textContent.replace('$',''));
      currentTip = Math.round(subtotal * (percent / 100) * 100) / 100;
      updateTotal();
    }

    function updateTotal() {
      const subtotal = parseFloat(document.getElementById('subtotal-display').textContent.replace('$',''));
      const tipDisplay = currentTip > 0 ? `$${currentTip.toFixed(2)}` : '$0.00';
      const total = subtotal + currentTip;
      const fee = Math.round(subtotal * 0.03 * 100) / 100;
      document.getElementById('tip-display').textContent = tipDisplay;
      document.getElementById('fee-amount').textContent = `$${fee.toFixed(2)}`;
      document.getElementById('total-display').textContent = `$${(total + fee).toFixed(2)}`;
      document.getElementById('final-total').textContent = `$${(total + fee).toFixed(2)}`;
    }

      document.querySelectorAll('#screen > div').forEach(d => d.classList.add('hidden'));

    }

    function updateLivePrice(input, service, size) {
      const value = parseInt(input.value) || 0;
      if (!db.landscaperPrices[selectedLandscaper]) db.landscaperPrices[selectedLandscaper] = {};
