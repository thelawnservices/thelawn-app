    function selectTime(el) {
      document.querySelectorAll('.time-slot').forEach(s => s.classList.remove('bg-[#34FF7A]', 'text-black'));
      el.classList.add('bg-[#34FF7A]', 'text-black');
    }

    function selectTip(el) {
      document.querySelectorAll('#pay-screen button').forEach(b => b.classList.remove('bg-[#34FF7A]', 'text-black'));
      el.classList.add('bg-[#34FF7A]', 'text-black');
    }

    function authorizeEscrowHold() {
      alert("✅ Payment held in escrow – both parties must confirm completion.");
      switchTab(0);
    }

    function switchTab(tab) {
      document.querySelectorAll('#screen > div').forEach(d => d.classList.add('hidden'));
      const screens = ['home-screen','search-screen','appointments-screen','myjobs-screen','profile-screen'];
      document.getElementById(screens[tab]).classList.remove('hidden');
      document.querySelectorAll('.nav-item').forEach((item,i) => i === tab ? item.classList.add('active','text-[#34FF7A]') : item.classList.remove('active','text-[#34FF7A]'));
    }

    function startBooking() { switchTab(2); }
