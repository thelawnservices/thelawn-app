  <script>
    function switchTab(tab) {
      document.querySelectorAll('#screen > div').forEach(d => d.classList.add('hidden'));
      const ids = ['home-screen','search-screen','appointments-screen','myjobs-screen','profile-screen'];
      document.getElementById(ids[tab]).classList.remove('hidden');
      document.querySelectorAll('.nav-item').forEach((item,i) => i === tab ? item.classList.add('active','text-[#34C759]') : item.classList.remove('active','text-[#34C759]'));
    }

    function toggleDeviceMode() {
      const c = document.getElementById('preview-container');
      c.classList.toggle('tablet');
      document.getElementById('mode-badge').textContent = c.classList.contains('tablet') ? 'TABLET' : 'PHONE';
    }

    function processPayment() {
      document.getElementById('pay-screen').classList.add('hidden');
      document.getElementById('processing-screen').classList.remove('hidden');
      setTimeout(() => {
        document.getElementById('processing-screen').classList.add('hidden');
        document.getElementById('success-screen').classList.remove('hidden');
      }, 1800);
    }

    function goToMyJobs() {
      document.getElementById('success-screen').classList.add('hidden');
      switchTab(3);
    }

    function toggleNotifications() {
      alert("🛎️ Notifications panel would open here (appointment reminders, on-the-way alerts, etc.)");
    }

    window.onload = () => {
      console.log("%c✅ COMPLETE The Lawn App Ready! All features included 🌱", "color:#34C759;font-weight:bold");
      switchTab(0);
    };
