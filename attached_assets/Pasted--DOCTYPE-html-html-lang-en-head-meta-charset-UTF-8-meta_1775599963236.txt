    /* REFINED NOTIFICATION FILTERING LOGIC */
    function sendRestrictedNotification(type, message, jobId) {
      const customerTypes = ['request_accepted', 'chat_from_landscaper', 'payment_sent', 'landscaper_arriving', 'service_complete'];
      const landscaperTypes = ['customer_chat', 'payment_received', 'customer_approved_complete', 'customer_booking'];

      if ((currentUserType === 'customer' && !customerTypes.includes(type)) ||
          (currentUserType === 'landscaper' && !landscaperTypes.includes(type))) {
        return;
      }

      const notif = {
        id: Date.now(),
        type,
        message,
        jobId,
        timestamp: new Date().toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'}),
        read: false
      };
      db.notifications.unshift(notif);
      saveDB();
    }

    /* LANDSCAPER ADDRESS CLICK → REAL GEOLOCATION + MAPS */
    function openMaps(address) {
      if (currentUserType !== 'landscaper') return;
      if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(pos => {
          const userLat = pos.coords.latitude;
          const userLon = pos.coords.longitude;
          const encoded = encodeURIComponent(address);
          const url = navigator.userAgent.match(/iPhone|iPad/i)
            ? `https://maps.apple.com/?saddr=${userLat},${userLon}&daddr=${encoded}`
            : `https://www.google.com/maps/dir/${userLat},${userLon}/${encoded}`;
          window.open(url, '_blank');
        }, () => {
          window.open(`https://www.google.com/maps/dir/?api=1&destination=${encodeURIComponent(address)}`, '_blank');
        });
      } else {
        window.open(`https://www.google.com/maps/dir/?api=1&destination=${encodeURIComponent(address)}`, '_blank');
      }
    }
