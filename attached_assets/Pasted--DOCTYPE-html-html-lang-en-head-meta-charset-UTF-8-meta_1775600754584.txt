  <script>
    let db = { 
      users: {}, 
      jobs: [], 
      requests: [], 
      notifications: [], 
      reviews: [],
      servicePhotos: [],
      landscaperPrices: {} 
    };
    let currentUserType = null;
    let currentUser = null;
    let currentRequestFiles = [];
    let currentChatJobId = null;
    let selectedServiceType = null;
    let selectedYardSize = null;

    function loadDB() { const saved = localStorage.getItem('thelawn_db'); if(saved) db = JSON.parse(saved); }
    function saveDB() { localStorage.setItem('thelawn_db', JSON.stringify(db)); }

    function startNewRequest() {
      const specificNeeds = document.getElementById('new-request-specific-needs').value.trim();
      const budget = parseFloat(document.getElementById('customer-budget-input').value) || 0;

      if (!selectedServiceType || !selectedYardSize || budget <= 0) {
        alert("Please select service type, yard size, and enter your budget amount.");
        return;
      }

      // Create the request with customer's custom price
      const newRequest = {
        id: Date.now(),
        customer: currentUser.first + " " + currentUser.last,
        service: selectedServiceType,
        yardSize: selectedYardSize,
        description: specificNeeds,
        customerPrice: budget,
        status: "Pending",
        address: currentUser.address || "Unknown",
        date: "Flexible",
        time: "Anytime"
      };
      db.requests.push(newRequest);
      saveDB();

      alert(`✅ Request submitted!\nService: ${selectedServiceType}\nBudget: $${budget}\nWaiting for landscaper acceptance.`);

      // Reset form
      document.getElementById('new-request-specific-needs').value = '';
      document.getElementById('customer-budget-input').value = '';
      document.getElementById('request-file-preview').innerHTML = '';
      currentRequestFiles = [];
      selectedServiceType = null;
      selectedYardSize = null;

      updateMyRequestsScreen();
    }

    function selectServiceType(btn) {
      document.querySelectorAll('.service-type-btn').forEach(b => b.classList.remove('bg-[#34FF7A]', 'text-black'));
      btn.classList.add('bg-[#34FF7A]', 'text-black');
      selectedServiceType = btn.dataset.type;
    }

    function selectYardSize(btn) {
      document.querySelectorAll('.yard-size-btn').forEach(b => b.classList.remove('bg-[#34FF7A]', 'text-black'));
      btn.classList.add('bg-[#34FF7A]', 'text-black');
      selectedYardSize = btn.dataset.size;
    }

    function updateMyRequestsScreen() {
      const container = document.getElementById('customer-requests-list');
      const myRequests = db.requests.filter(r => r.customer === (currentUser.first + " " + currentUser.last));
      let html = '';
      if (myRequests.length) {
        html = myRequests.map(r => `
          <div class="card rounded-3xl p-5">
            <p class="font-medium">${r.service}</p>
            <p class="text-xs text-white/70">${r.yardSize} yard • $${r.customerPrice}</p>
            <p class="text-xs text-white/70">${r.description}</p>
            <p class="text-[#34FF7A] text-sm mt-2">${r.status}</p>
          </div>
        `).join('');
      } else {
        html = `<p class="text-white/50 text-center py-8">No requests yet.</p>`;
      }
      container.innerHTML = html;
    }

    window.onload = () => {
      loadDB();
      console.log("%c✅ The Lawn – Customer request form now includes custom 'Amount Willing to Pay' field. Price is honored on acceptance 🌱", "color:#34FF7A;font-weight:bold");
    };
  </script>
