    function updateAppointmentsScreen() {
      const container = document.getElementById('appointments-list');
      container.innerHTML = '<p class="text-xs text-white/70 mb-3">Upcoming Appointments</p>';

      const futureJobs = db.jobs.filter(j => 
        j.status === "Future" && 
        ((currentUserType === 'landscaper' && j.landscaper === currentUser.businessName) || 
         (currentUserType === 'customer' && j.customer === (currentUser.first + " " + currentUser.last)))
      );

      if (futureJobs.length === 0) {
        container.innerHTML += `<p class="text-white/50 text-center py-8">No upcoming appointments</p>`;
        return;
      }

      futureJobs.forEach(job => {
        const div = document.createElement('div');
        div.className = `card rounded-3xl p-5`;
        div.innerHTML = `<p class="font-medium">${job.service}</p><p class="text-xs text-white/70">${job.date} at ${job.time}</p>`;
        container.appendChild(div);
      });
    }

    function updateMyJobsScreen() {
      const activeContainer = document.getElementById('active-jobs-list');
      const completedContainer = document.getElementById('completed-jobs-list');
      activeContainer.innerHTML = '';
      completedContainer.innerHTML = '';

      const myJobs = db.jobs.filter(j => 
        (currentUserType === 'customer' && j.customer === (currentUser.first + " " + currentUser.last)) ||
        (currentUserType === 'landscaper' && j.landscaper === currentUser.businessName)
      );

      // ACTIVE JOBS (In Progress / Active) – only for both accounts
      const activeJobs = myJobs.filter(j => ['In Progress', 'Active', 'Arrived'].includes(j.status));
      if (activeJobs.length === 0) {
        activeContainer.innerHTML = `<p class="text-white/50 text-center py-8">No active jobs right now</p>`;
      } else {
        activeJobs.forEach(job => {
          const div = document.createElement('div');
          div.className = `card rounded-3xl p-5`;
          div.innerHTML = `
            <div class="flex justify-between items-start">
              <div>
                <p class="font-semibold">${job.service}</p>
                <p class="text-xs text-white/70">${job.date} • ${job.time}</p>
